31      @Test
32      void testFuzzyLikeBothPresent() {
33          String fName = "Fred";
34          String lName = "Flintstone";
35  
36          SelectStatementProvider selectStatement = select(id, firstName, lastName)
37                  .from(person)
38                  .where(firstName, isLike(fName).filter(Objects::nonNull).map(s -> "%" + s + "%"))
39                  .and(lastName, isLike(lName).filter(Objects::nonNull).map(s -> "%" + s + "%"))
40                  .build()
41                  .render(RenderingStrategies.MYBATIS3);
42  
43          String expected = "select person_id, first_name, last_name"
44                  + " from Person"
45                  + " where first_name like #{parameters.p1}"
46                  + " and last_name like #{parameters.p2}";
47  
48          assertThat(selectStatement.getSelectStatement()).isEqualTo(expected);
49          assertThat(selectStatement.getParameters()).containsEntry("p1", "%Fred%");
50          assertThat(selectStatement.getParameters()).containsEntry("p2", "%Flintstone%");
51      }
