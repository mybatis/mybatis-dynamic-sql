16  package org.mybatis.dynamic.sql.where.render;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  
20  import java.sql.JDBCType;
21  import java.util.HashMap;
22  import java.util.Map;
23  import java.util.concurrent.atomic.AtomicInteger;
24  
25  import org.junit.jupiter.api.Test;
26  import org.mybatis.dynamic.sql.SqlColumn;
27  import org.mybatis.dynamic.sql.SqlCriterion;
28  import org.mybatis.dynamic.sql.SqlTable;
29  import org.mybatis.dynamic.sql.render.RenderingStrategies;
30  import org.mybatis.dynamic.sql.render.TableAliasCalculator;
31  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
32  import org.mybatis.dynamic.sql.where.condition.IsEqualTo;
33  
34  public class CriterionRendererTest {
35  
36      @Test
37      public void testAliasWithIgnore() {
38          SqlTable table = SqlTable.of("foo");
39          SqlColumn<Integer> column = table.column("id", JDBCType.INTEGER);
40          
41          IsEqualTo<Integer> condition = IsEqualTo.of(() -> 3);
42          SqlCriterion<Integer> criterion = SqlCriterion.withColumn(column)
43                  .withCondition(condition)
44                  .build();
45          AtomicInteger sequence = new AtomicInteger(1);
46          FragmentAndParameters fp = CriterionRenderer.withCriterion(criterion)
47                  .withSequence(sequence)
48                  .withRenderingStrategy(RenderingStrategies.MYBATIS3)
49                  .withTableAliasCalculator(TableAliasCalculator.empty())
50                  .build()
51                  .render()
52                  .get()
53                  .renderWithInitialConnector();
54          
55          assertThat(fp.fragment()).isEqualTo("id = #{parameters.p1,jdbcType=INTEGER}");
56          assertThat(fp.parameters().size()).isEqualTo(1);
57          assertThat(fp.parameters().get("p1")).isEqualTo(3);
58      }
59  
60      @Test
61      public void testAliasWithoutIgnore() {
62          SqlTable table = SqlTable.of("foo");
63          SqlColumn<Integer> column = table.column("id", JDBCType.INTEGER);
64          IsEqualTo<Integer> condition = IsEqualTo.of(() -> 3);
65          SqlCriterion<Integer> criterion = SqlCriterion.withColumn(column)
66                  .withCondition(condition)
67                  .build();
68          AtomicInteger sequence = new AtomicInteger(1);
69          
70          Map<SqlTable, String> tableAliases = new HashMap<>();
71          tableAliases.put(table, "a");
72          
73          FragmentAndParameters fp = CriterionRenderer.withCriterion(criterion)
74                  .withSequence(sequence)
75                  .withRenderingStrategy(RenderingStrategies.MYBATIS3)
76                  .withTableAliasCalculator(TableAliasCalculator.of(tableAliases))
77                  .build()
78                  .render()
79                  .get()
80                  .renderWithInitialConnector();
81          
82          assertThat(fp.fragment()).isEqualTo("a.id = #{parameters.p1,jdbcType=INTEGER}");
83          assertThat(fp.parameters().size()).isEqualTo(1);
84          assertThat(fp.parameters().get("p1")).isEqualTo(3);
85      }
86  }
