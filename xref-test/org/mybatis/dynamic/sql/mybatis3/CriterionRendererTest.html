38  class CriterionRendererTest {
39      @Test
40      void testAliasWithIgnore() {
41          SqlTable table = SqlTable.of("foo");
42          SqlColumn<Integer> column = table.column("id", JDBCType.INTEGER);
43  
44          IsEqualTo<Integer> condition = SqlBuilder.isEqualTo(() -> 3);
45          ColumnAndConditionCriterion<Integer> criterion = ColumnAndConditionCriterion.withColumn(column)
46                  .withCondition(condition)
47                  .build();
48          AtomicInteger sequence = new AtomicInteger(1);
49  
50          CriterionRenderer renderer = new CriterionRenderer.Builder()
51                  .withSequence(sequence)
52                  .withRenderingStrategy(RenderingStrategies.MYBATIS3)
53                  .withTableAliasCalculator(TableAliasCalculator.empty())
54                  .build();
55  
56          assertThat(criterion.accept(renderer)).hasValueSatisfying(rc -> {
57              FragmentAndParameters fp = rc.fragmentAndParametersWithConnector();
58              assertThat(fp.fragment()).isEqualTo("id = #{parameters.p1,jdbcType=INTEGER}");
59              assertThat(fp.parameters()).hasSize(1);
60          });
61      }
