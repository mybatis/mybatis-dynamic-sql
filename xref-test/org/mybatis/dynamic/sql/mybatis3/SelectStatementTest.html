34  @RunWith(JUnitPlatform.class)
35  public class SelectStatementTest {
36      public static final SqlTable table = SqlTable.of("foo");
37      public static final SqlColumn<Date> column1 = table.column("column1", JDBCType.DATE);
38      public static final SqlColumn<Integer> column2 = table.column("column2", JDBCType.INTEGER);
39  
40      @Test
41      public void testSimpleCriteriaWithoutAlias() {
42          Date d = new Date();
43  
44          SelectStatement selectStatement = select(column1, column2)
45                  .from(table, "a")
46                  .where(column1, isEqualTo(d))
47                  .or(column2, isEqualTo(4))
48                  .and(column2, isLessThan(3))
49                  .build()
50                  .render(RenderingStrategy.MYBATIS3);
51  
52          assertThat(selectStatement.getSelectStatement()).isEqualTo(
53                  "select a.column1, a.column2 from foo a where a.column1 = #{parameters.p1,jdbcType=DATE} or a.column2 = #{parameters.p2,jdbcType=INTEGER} and a.column2 < #{parameters.p3,jdbcType=INTEGER}");
54  
55          Map<String, Object> parameters = selectStatement.getParameters();
56          assertThat(parameters.get("p1")).isEqualTo(d);
57          assertThat(parameters.get("p2")).isEqualTo(4);
58          assertThat(parameters.get("p3")).isEqualTo(3);
59      }
60  
61      @Test
62      public void testComplexCriteriaWithoutAlias() {
63          Date d = new Date();
64  
65          SelectStatement selectStatement = select(column1, column2)
66                  .from(table, "a")
67                  .where(column1, isEqualTo(d))
68                  .or(column2, isEqualTo(4))
69                  .and(column2, isLessThan(3))
70                  .or(column2, isEqualTo(4), and(column2, isEqualTo(6)))
71                  .and(column2, isLessThan(3), or(column1, isEqualTo(d)))
72                  .build()
73                  .render(RenderingStrategy.MYBATIS3);
74          
75  
76          String expected = "select a.column1, a.column2 "
77                  + "from foo a " 
78                  + "where a.column1 = #{parameters.p1,jdbcType=DATE}"
79                  + " or a.column2 = #{parameters.p2,jdbcType=INTEGER}"
80                  + " and a.column2 < #{parameters.p3,jdbcType=INTEGER}"
81                  + " or (a.column2 = #{parameters.p4,jdbcType=INTEGER} and a.column2 = #{parameters.p5,jdbcType=INTEGER})"
82                  + " and (a.column2 < #{parameters.p6,jdbcType=INTEGER} or a.column1 = #{parameters.p7,jdbcType=DATE})";
83          
84          SoftAssertions.assertSoftly(softly -> {
85              softly.assertThat(selectStatement.getSelectStatement()).isEqualTo(expected);
86  
87              Map<String, Object> parameters = selectStatement.getParameters();
88              softly.assertThat(parameters.get("p1")).isEqualTo(d);
89              softly.assertThat(parameters.get("p2")).isEqualTo(4);
90              softly.assertThat(parameters.get("p3")).isEqualTo(3);
91              softly.assertThat(parameters.get("p4")).isEqualTo(4);
92              softly.assertThat(parameters.get("p5")).isEqualTo(6);
93              softly.assertThat(parameters.get("p6")).isEqualTo(3);
94              softly.assertThat(parameters.get("p7")).isEqualTo(d);
95          });
96      }

121     @Test
122     public void testComplexCriteriaWithAlias() {
123         Date d = new Date();
124 
125         SelectStatement selectStatement = select(column1, column2)
126                 .from(table, "a")
127                 .where(column1, isEqualTo(d))
128                 .or(column2, isEqualTo(4))
129                 .and(column2, isLessThan(3))
130                 .or(column2, isEqualTo(4), and(column2, isEqualTo(6), or(column2, isEqualTo(7))))
131                 .and(column2, isLessThan(3), or(column1, isEqualTo(d), and(column2, isEqualTo(88))))
132                 .build()
133                 .render(RenderingStrategy.MYBATIS3);
134         
135 
136         String expected = "select a.column1, a.column2 "
137                 + "from foo a "
138                 + "where a.column1 = #{parameters.p1,jdbcType=DATE}"
139                 + " or a.column2 = #{parameters.p2,jdbcType=INTEGER}"
140                 + " and a.column2 < #{parameters.p3,jdbcType=INTEGER}"
141                 + " or (a.column2 = #{parameters.p4,jdbcType=INTEGER} and (a.column2 = #{parameters.p5,jdbcType=INTEGER} or a.column2 = #{parameters.p6,jdbcType=INTEGER}))"
142                 + " and (a.column2 < #{parameters.p7,jdbcType=INTEGER} or (a.column1 = #{parameters.p8,jdbcType=DATE} and a.column2 = #{parameters.p9,jdbcType=INTEGER}))";
143         
144         SoftAssertions.assertSoftly(softly -> {
145             softly.assertThat(selectStatement.getSelectStatement()).isEqualTo(expected);
146 
147             Map<String, Object> parameters = selectStatement.getParameters();
148             softly.assertThat(parameters.get("p1")).isEqualTo(d);
149             softly.assertThat(parameters.get("p2")).isEqualTo(4);
150             softly.assertThat(parameters.get("p3")).isEqualTo(3);
151             softly.assertThat(parameters.get("p4")).isEqualTo(4);
152             softly.assertThat(parameters.get("p5")).isEqualTo(6);
153             softly.assertThat(parameters.get("p6")).isEqualTo(7);
154             softly.assertThat(parameters.get("p7")).isEqualTo(3);
155             softly.assertThat(parameters.get("p8")).isEqualTo(d);
156             softly.assertThat(parameters.get("p9")).isEqualTo(88);
157         });
158     }
