16  package examples.custom_render;
17  
18  import java.util.List;
19  import java.util.Optional;
20  
21  import org.apache.ibatis.annotations.Result;
22  import org.apache.ibatis.annotations.ResultMap;
23  import org.apache.ibatis.annotations.Results;
24  import org.apache.ibatis.annotations.SelectProvider;
25  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
26  import org.mybatis.dynamic.sql.util.SqlProviderAdapter;
27  import org.mybatis.dynamic.sql.util.mybatis3.CommonDeleteMapper;
28  import org.mybatis.dynamic.sql.util.mybatis3.CommonInsertMapper;
29  import org.mybatis.dynamic.sql.util.mybatis3.CommonSelectMapper;
30  import org.mybatis.dynamic.sql.util.mybatis3.CommonUpdateMapper;
31  
32  public interface JsonTestMapper extends CommonDeleteMapper, CommonInsertMapper<JsonTestRecord>, CommonSelectMapper,
33          CommonUpdateMapper {
34      @SelectProvider(type = SqlProviderAdapter.class, method = "select")
35      @Results(id = "JsonTestResult", value = {
36              @Result(column = "id", property = "id", id = true),
37              @Result(column = "description", property = "description"),
38              @Result(column = "info", property = "info")
39      })
40      List<JsonTestRecord> selectMany(SelectStatementProvider selectStatement);
41  
42      @SelectProvider(type = SqlProviderAdapter.class, method = "select")
43      @ResultMap("JsonTestResult")
44      Optional<JsonTestRecord> selectOne(SelectStatementProvider selectStatement);
45  }
