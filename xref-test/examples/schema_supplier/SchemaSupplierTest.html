35  import org.apache.ibatis.transaction.jdbc.JdbcTransactionFactory;

37  import org.junit.jupiter.api.Test;
38  import org.mybatis.dynamic.sql.select.SelectDSLCompleter;
39  
40  public class SchemaSupplierTest {
41  
42      private static final String JDBC_URL = "jdbc:hsqldb:mem:aname";
43      private static final String JDBC_DRIVER = "org.hsqldb.jdbcDriver";
44  
45      private SqlSessionFactory sqlSessionFactory;
46  
47      @BeforeEach
48      public void setup() throws Exception {
49          Class.forName(JDBC_DRIVER);
50          InputStream is = getClass().getResourceAsStream("/examples/schema_supplier/CreateDB.sql");
51          try (Connection connection = DriverManager.getConnection(JDBC_URL, "sa", "")) {
52              ScriptRunner sr = new ScriptRunner(connection);
53              sr.setLogWriter(null);
54              sr.runScript(new InputStreamReader(is));
55          }
56  
57          UnpooledDataSource ds = new UnpooledDataSource(JDBC_DRIVER, JDBC_URL, "sa", "");
58          Environment environment = new Environment("test", new JdbcTransactionFactory(), ds);
59          Configuration config = new Configuration(environment);
60          config.addMapper(UserMapper.class);
61          sqlSessionFactory = new SqlSessionFactoryBuilder().build(config);
62      }
63  
64      @Test
65      public void testUnsetSchemaProperty() {
66          System.clearProperty(SchemaSupplier.schema_property);
67  
68          try (SqlSession session = sqlSessionFactory.openSession()) {
69              UserMapper mapper = session.getMapper(UserMapper.class);
70  
71              assertThrows(PersistenceException.class, () -> {
72                  Userhtml#User">User user = new User();
73                  user.setId(1);
74                  user.setName("Fred");
75  
76                  int rows = mapper.insert(user);
77  
78                  assertThat(rows).isEqualTo(1);
79              });
80          }
81      }
82  
83      @Test
84      public void testSchemaProperty() {
85          System.setProperty(SchemaSupplier.schema_property, "schema1");
86  
87          try (SqlSession session = sqlSessionFactory.openSession()) {
88              UserMapper mapper = session.getMapper(UserMapper.class);
89  
90              insertFlintstones(mapper);
91              
92              List<User> records = mapper.select(SelectDSLCompleter.allRows());
93              assertThat(records.size()).isEqualTo(2);
94          }
95      }
96  
97      @Test
98      public void testSchemaSwitchingProperty() {
99          try (SqlSession session = sqlSessionFactory.openSession()) {
100             UserMapper mapper = session.getMapper(UserMapper.class);
101 
102             System.setProperty(SchemaSupplier.schema_property, "schema1");
103             insertFlintstones(mapper);
104             
105             List<User> records = mapper.select(SelectDSLCompleter.allRows());
106             assertThat(records.size()).isEqualTo(2);
107 
108             
109             System.setProperty(SchemaSupplier.schema_property, "schema2");
110             insertRubbles(mapper);
111             
112             records = mapper.select(SelectDSLCompleter.allRows());
113             assertThat(records.size()).isEqualTo(3);
114         }
115     }
116 
117     private void insertFlintstones(UserMapper mapper) {
118         Userhtml#User">User user = new User();
119         user.setId(1);
120         user.setName("Fred");
121         int rows = mapper.insert(user);
122         assertThat(rows).isEqualTo(1);
123 
124         user = new User();
125         user.setId(2);
126         user.setName("Wilma");
127         rows = mapper.insert(user);
128         assertThat(rows).isEqualTo(1);
129     }
130 
131     private void insertRubbles(UserMapper mapper) {
132         Userhtml#User">User user = new User();
133         user.setId(1);
134         user.setName("Barney");
135         int rows = mapper.insert(user);
136         assertThat(rows).isEqualTo(1);
137 
138         user = new User();
139         user.setId(2);
140         user.setName("Betty");
141         rows = mapper.insert(user);
142         assertThat(rows).isEqualTo(1);
143 
144         user = new User();
145         user.setId(3);
146         user.setName("Bamm Bamm");
147         rows = mapper.insert(user);
148         assertThat(rows).isEqualTo(1);
149     }
150 }
