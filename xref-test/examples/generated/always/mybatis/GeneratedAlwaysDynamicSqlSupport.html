16  package examples.generated.always.mybatis;
17  
18  import static org.mybatis.dynamic.sql.SqlBuilder.*;
19  
20  import java.sql.JDBCType;
21  
22  import org.mybatis.dynamic.sql.SqlColumn;
23  import org.mybatis.dynamic.sql.SqlTable;
24  import org.mybatis.dynamic.sql.insert.render.InsertStatement;
25  import org.mybatis.dynamic.sql.render.RenderingStrategy;
26  import org.mybatis.dynamic.sql.select.QueryExpressionDSL;
27  import org.mybatis.dynamic.sql.select.SelectModel;
28  import org.mybatis.dynamic.sql.select.render.SelectStatement;
29  import org.mybatis.dynamic.sql.update.UpdateDSL;
30  import org.mybatis.dynamic.sql.update.UpdateModel;
31  import org.mybatis.dynamic.sql.update.render.UpdateStatement;
32  
33  import examples.generated.always.GeneratedAlwaysRecord;
34  
35  public final class GeneratedAlwaysDynamicSqlSupport {
36      public static final GeneratedAlways generatedAlways = new GeneratedAlways();
37      public static final SqlColumn<Integer> id = generatedAlways.id;
38      public static final SqlColumn<String> firstName = generatedAlways.firstName;
39      public static final SqlColumn<String> lastName = generatedAlways.lastName;
40      public static final SqlColumn<String> fullName = generatedAlways.fullName;
41      
42      public static final class GeneratedAlways extends SqlTable {
43          public final SqlColumn<Integer> id = column("id", JDBCType.INTEGER);
44          public final SqlColumn<String> firstName = column("first_name", JDBCType.VARCHAR);
45          public final SqlColumn<String> lastName = column("last_name", JDBCType.VARCHAR);
46          public final SqlColumn<String> fullName = column("full_name", JDBCType.VARCHAR);
47  
48          public GeneratedAlways() {
49              super("GeneratedAlways");
50          }
51      }
52      
53      public static InsertStatement<GeneratedAlwaysRecord> buildInsert(GeneratedAlwaysRecord record) {
54          return insert(record)
55                  .into(generatedAlways)
56                  .map(id).toProperty("id")
57                  .map(firstName).toProperty("firstName")
58                  .map(lastName).toProperty("lastName")
59                  .build()
60                  .render(RenderingStrategy.MYBATIS3);
61      }
62  
63      public static InsertStatement<GeneratedAlwaysRecord> buildInsertSelectiveStatement(GeneratedAlwaysRecord record) {
64          return insert(record)
65                  .into(generatedAlways)
66                  .map(id).toPropertyWhenPresent("id")
67                  .map(firstName).toPropertyWhenPresent("firstName")
68                  .map(lastName).toPropertyWhenPresent("lastName")
69                  .build()
70                  .render(RenderingStrategy.MYBATIS3);
71      }
72      
73      public static UpdateStatement buildUpdateByPrimaryKeyStatement(GeneratedAlwaysRecord record) {
74          return update(generatedAlways)
75                  .set(firstName).equalTo(record.getFirstName())
76                  .set(lastName).equalTo(record.getLastName())
77                  .where(id, isEqualTo(record.getId()))
78                  .build()
79                  .render(RenderingStrategy.MYBATIS3);
80      }
81  
82      public static UpdateStatement buildUpdateByPrimaryKeySelectiveStatement(GeneratedAlwaysRecord record) {
83          return update(generatedAlways)
84                  .set(firstName).equalToWhenPresent(record.getFirstName())
85                  .set(lastName).equalToWhenPresent(record.getLastName())
86                  .where(id, isEqualTo(record.getId()))
87                  .build()
88                  .render(RenderingStrategy.MYBATIS3);
89      }
90  
91      public static UpdateDSL<UpdateModel> updateByExample(GeneratedAlwaysRecord record) {
92          return update(generatedAlways)
93                  .set(id).equalTo(record.getId())
94                  .set(firstName).equalTo(record.getFirstName())
95                  .set(lastName).equalTo(record.getLastName());
96      }
97  
98      public static UpdateDSL<UpdateModel> updateByExampleSelective(GeneratedAlwaysRecord record) {
99          return update(generatedAlways)
100                 .set(id).equalToWhenPresent(record.getId())
101                 .set(firstName).equalToWhenPresent(record.getFirstName())
102                 .set(lastName).equalToWhenPresent(record.getLastName());
103     }
104     
105     public static QueryExpressionDSL<SelectModel> selectByExample() {
106         return select(id.as("A_ID"), firstName, lastName, fullName)
107                 .from(generatedAlways, "a");
108     }
109     
110     public static SelectStatement selectByPrimaryKey(int id_) {
111         return select(id.as("A_ID"), firstName, lastName, fullName)
112                 .from(generatedAlways)
113                 .where(id, isEqualTo(id_))
114                 .build()
115                 .render(RenderingStrategy.MYBATIS3);
116     }
117 }
