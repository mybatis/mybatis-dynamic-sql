44  class PersonMapperTest {
45  
46      private static final String JDBC_URL = "jdbc:hsqldb:mem:aname";
47      private static final String JDBC_DRIVER = "org.hsqldb.jdbcDriver";
48  
49      private SqlSessionFactory sqlSessionFactory;
50  
51      @BeforeEach
52      void setup() throws Exception {
53          Class.forName(JDBC_DRIVER);
54          InputStream is = getClass().getResourceAsStream("/examples/generated/always/CreateGeneratedAlwaysDB.sql");
55          try (Connection connection = DriverManager.getConnection(JDBC_URL, "sa", "")) {
56              ScriptRunner sr = new ScriptRunner(connection);
57              sr.setLogWriter(null);
58              sr.runScript(new InputStreamReader(is));
59          }
60  
61          UnpooledDataSource ds = new UnpooledDataSource(JDBC_DRIVER, JDBC_URL, "sa", "");
62          Environment environment = new Environment("test", new JdbcTransactionFactory(), ds);
63          Configuration config = new Configuration(environment);
64          config.addMapper(PersonMapper.class);
65          sqlSessionFactory = new SqlSessionFactoryBuilder().build(config);
66      }
67  
68      @Test
69      void testInsertSelectWithOneRecord() {
70          try (SqlSession session = sqlSessionFactory.openSession()) {
71              PersonMapper mapper = session.getMapper(PersonMapper.class);
72  
73              PersonRecord record = new PersonRecord();
74              record.setFirstName("Fred");
75              record.setLastName("Flintstone");
76  
77              int rows = mapper.insert(record);
78  
79              assertThat(rows).isEqualTo(1);
80              assertThat(record.getId()).isEqualTo(22);
81  
82              InsertSelectStatementProvider insertSelectStatement = SqlBuilder.insertInto(person)
83                      .withColumnList(firstName, lastName)
84                      .withSelectStatement(SqlBuilder.select(firstName, lastName).from(person))
85                      .build()
86                      .render(RenderingStrategies.MYBATIS3);
87  
88              rows = mapper.insertSelect(insertSelectStatement);
89              assertThat(rows).isEqualTo(1);
90              assertThat(insertSelectStatement.getParameters()).containsEntry("id", 23);
91  
92              List<PersonRecord> records = mapper.select(c -> c.orderBy(id));
93              assertThat(records).hasSize(2);
94              assertThat(records.get(0).getId()).isEqualTo(22);
95              assertThat(records.get(1).getId()).isEqualTo(23);
96          }
97      }
98  
99      @Test
100     void testInsertSelectWithMultipleRecords() {
101         try (SqlSession session = sqlSessionFactory.openSession()) {
102             PersonMapper mapper = session.getMapper(PersonMapper.class);
103 
104             PersonRecord record1 = new PersonRecord();
105             record1.setFirstName("Fred");
106             record1.setLastName("Flintstone");
107 
108             PersonRecord record2 = new PersonRecord();
109             record2.setFirstName("Barney");
110             record2.setLastName("Rubble");
111 
112             int rows = mapper.insertMultiple(record1, record2);
113 
114             assertThat(rows).isEqualTo(2);
115             assertThat(record1.getId()).isEqualTo(22);
116             assertThat(record2.getId()).isEqualTo(23);
117 
118             InsertSelectStatementProvider insertSelectStatement = SqlBuilder.insertInto(person)
119                     .withColumnList(firstName, lastName)
120                     .withSelectStatement(SqlBuilder.select(firstName, lastName).from(person))
121                     .build()
122                     .render(RenderingStrategies.MYBATIS3);
123 
124             insertSelectStatement.getParameters().put("keys", new ArrayList<GeneratedKey>());
125 
126             GeneratedKeyList keys = new GeneratedKeyList(5);
127 
128             rows = mapper.insertSelect(insertSelectStatement, keys);
129             assertThat(rows).isEqualTo(2);
130             assertThat(keys.get(0).getKey()).isEqualTo(24);
131             assertThat(keys.get(1).getKey()).isEqualTo(25);
132 
133             List<PersonRecord> records = mapper.select(c -> c.orderBy(id));
134             assertThat(records).hasSize(4);
135             assertThat(records.get(0).getId()).isEqualTo(22);
136             assertThat(records.get(1).getId()).isEqualTo(23);
137             assertThat(records.get(2).getId()).isEqualTo(24);
138             assertThat(records.get(3).getId()).isEqualTo(25);
139         }
140     }
