52  @RunWith(JUnitPlatform.class)
53  public class SpringTest {
54      private EmbeddedDatabase db;
55      private NamedParameterJdbcTemplate template;
56      
57      @BeforeEach
58      public void setup() {
59          db = new EmbeddedDatabaseBuilder()
60                  .setType(EmbeddedDatabaseType.HSQL)
61                  .generateUniqueName(true)
62                  .addScript("classpath:/examples/generated/always/CreateGeneratedAlwaysDB.sql")
63                  .build();
64          template = new NamedParameterJdbcTemplate(db);
65      }
66      
67      @Test
68      public void testRender() {
69          SelectStatement selectStatement = select(id.as("A_ID"), firstName, lastName, fullName)
70                  .from(generatedAlways, "a")
71                  .where(id, isGreaterThan(3))
72                  .orderBy(id.descending())
73                  .build()
74                  .render(RenderingStrategy.SPRING_NAMED_PARAMETER);
75          
76          String expected = "select a.id as A_ID, a.first_name, a.last_name, a.full_name "
77                  + "from GeneratedAlways a "
78                  + "where a.id > :p1 "
79                  + "order by id DESC";
80          
81          assertThat(selectStatement.getSelectStatement()).isEqualTo(expected);
82      }
83      
84      @Test
85      public void testSelect() {
86          SelectStatement selectStatement = select(id, firstName, lastName, fullName)
87                  .from(generatedAlways)
88                  .where(id, isGreaterThan(3))
89                  .orderBy(id.descending())
90                  .build()
91                  .render(RenderingStrategy.SPRING_NAMED_PARAMETER);
92          
93          SqlParameterSource namedParameters = new MapSqlParameterSource(selectStatement.getParameters());
94          List<GeneratedAlwaysRecord> records = template.query(selectStatement.getSelectStatement(), namedParameters,
95                  new RowMapper<GeneratedAlwaysRecord>(){
96                      @Override
97                      public GeneratedAlwaysRecord mapRow(ResultSet rs, int rowNum) throws SQLException {
98                          GeneratedAlwaysRecord record = new GeneratedAlwaysRecord();
99                          record.setId(rs.getInt(1));
100                         record.setFirstName(rs.getString(2));
101                         record.setLastName(rs.getString(3));
102                         record.setFullName(rs.getString(4));
103                         return record;
104                     }
105                 });
106         
107         assertThat(records.size()).isEqualTo(3);
108         assertThat(records.get(0).getId()).isEqualTo(6);
109         assertThat(records.get(0).getFirstName()).isEqualTo("Bamm Bamm");
110         assertThat(records.get(0).getLastName()).isEqualTo("Rubble");
111         assertThat(records.get(0).getFullName()).isEqualTo("Bamm Bamm Rubble");
112         
113         assertThat(records.get(1).getId()).isEqualTo(5);
114         assertThat(records.get(2).getId()).isEqualTo(4);
115     }

117     @Test
118     public void testDelete() {
119         DeleteStatement deleteStatement = deleteFrom(generatedAlways)
120                 .where(id,  isLessThan(3))
121                 .build()
122                 .render(RenderingStrategy.SPRING_NAMED_PARAMETER);
123         
124         SqlParameterSource parameterSource = new MapSqlParameterSource(deleteStatement.getParameters());
125         
126         int rows = template.update(deleteStatement.getDeleteStatement(), parameterSource);
127         
128         assertThat(rows).isEqualTo(2);
129     }

131     @Test
132     public void testInsert() {
133         GeneratedAlwaysRecord record = new GeneratedAlwaysRecord();
134         record.setId(100);
135         record.setFirstName("Bob");
136         record.setLastName("Jones");
137         
138         InsertStatement<GeneratedAlwaysRecord> insertStatement = insert(record)
139                 .into(generatedAlways)
140                 .map(id).toProperty("id")
141                 .map(firstName).toProperty("firstName")
142                 .map(lastName).toProperty("lastName")
143                 .build()
144                 .render(RenderingStrategy.SPRING_NAMED_PARAMETER);
145         
146         SqlParameterSource parameterSource = new BeanPropertySqlParameterSource(insertStatement.getRecord());
147         KeyHolder keyHolder = new GeneratedKeyHolder();
148         
149         int rows = template.update(insertStatement.getInsertStatement(), parameterSource, keyHolder);
150         String generatedKey = (String) keyHolder.getKeys().get("FULL_NAME");
151         
152         assertThat(rows).isEqualTo(1);
153         assertThat(generatedKey).isEqualTo("Bob Jones");
154     }
155     
156     @Test
157     public void testInsertBatch() {
158         List<GeneratedAlwaysRecord> records = new ArrayList<>();
159         GeneratedAlwaysRecord record = new GeneratedAlwaysRecord();
160         record.setId(100);
161         record.setFirstName("Bob");
162         record.setLastName("Jones");
163         records.add(record);
164         
165         record = new GeneratedAlwaysRecord();
166         record.setId(101);
167         record.setFirstName("Jim");
168         record.setLastName("Smith");
169         records.add(record);
170 
171         SqlParameterSource[] batch = SqlParameterSourceUtils.createBatch(records.toArray());
172         
173         BatchInsert<GeneratedAlwaysRecord> batchInsert = insert(records)
174                 .into(generatedAlways)
175                 .map(id).toProperty("id")
176                 .map(firstName).toProperty("firstName")
177                 .map(lastName).toProperty("lastName")
178                 .build()
179                 .render(RenderingStrategy.SPRING_NAMED_PARAMETER);
180         
181         int[] updateCounts = template.batchUpdate(batchInsert.getInsertStatementSQL(), batch);
182         
183         assertThat(updateCounts.length).isEqualTo(2);
184         assertThat(updateCounts[0]).isEqualTo(1);
185         assertThat(updateCounts[1]).isEqualTo(1);
186     }
187 
188     @Test
189     public void testUpdate() {
190         UpdateStatement updateStatement = update(generatedAlways)
191                 .set(firstName).equalToStringConstant("Rob")
192                 .where(id,  isIn(1, 5, 22))
193                 .build()
194                 .render(RenderingStrategy.SPRING_NAMED_PARAMETER);
195         
196         SqlParameterSource parameterSource = new MapSqlParameterSource(updateStatement.getParameters());
197         
198         int rows = template.update(updateStatement.getUpdateStatement(), parameterSource);
199         
200         assertThat(rows).isEqualTo(2);
201     }
