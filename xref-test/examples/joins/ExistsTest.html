62          Environment environment = new Environment("test", new JdbcTransactionFactory(), ds);

66          sqlSessionFactory = new SqlSessionFactoryBuilder().build(config);

69      @Test
70      void testExists() {
71          try (SqlSession session = sqlSessionFactory.openSession()) {
72              CommonSelectMapper mapper = session.getMapper(CommonSelectMapper.class);
73  
74              SelectStatementProvider selectStatement = select(itemMaster.allColumns())
75                      .from(itemMaster, "im")
76                      .where(exists(
77                              select(orderLine.allColumns())
78                                      .from(orderLine, "ol")
79                                      .where(orderLine.itemId, isEqualTo(itemMaster.itemId.qualifiedWith("im")))
80                      ))
81                      .orderBy(itemMaster.itemId)
82                      .build()
83                      .render(RenderingStrategies.MYBATIS3);
84  
85              String expectedStatement = "select im.* from ItemMaster im"
86                      + " where exists (select ol.* from OrderLine ol where ol.item_id = im.item_id)"
87                      + " order by item_id";
88              assertThat(selectStatement.getSelectStatement()).isEqualTo(expectedStatement);
89  
90              List<Map<String, Object>> rows = mapper.selectManyMappedRows(selectStatement);
91  
92              assertThat(rows).hasSize(3);
93              Map<String, Object> row = rows.get(0);
94              assertThat(row).containsEntry("ITEM_ID", 22);
95              assertThat(row).containsEntry("DESCRIPTION", "Helmet");
96  
97              row = rows.get(1);
98              assertThat(row).containsEntry("ITEM_ID", 33);
99              assertThat(row).containsEntry("DESCRIPTION", "First Base Glove");
100 
101             row = rows.get(2);
102             assertThat(row).containsEntry("ITEM_ID", 44);
103             assertThat(row).containsEntry("DESCRIPTION", "Outfield Glove");
104         }
105     }
