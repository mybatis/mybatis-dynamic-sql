21  import org.apache.ibatis.annotations.Result;
22  import org.apache.ibatis.annotations.ResultMap;
23  import org.apache.ibatis.annotations.Results;
24  import org.apache.ibatis.annotations.Select;
25  import org.apache.ibatis.annotations.SelectProvider;
26  import org.apache.ibatis.session.RowBounds;
27  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
28  import org.mybatis.dynamic.sql.util.SqlProviderAdapter;
29  import org.mybatis.dynamic.sql.util.mybatis3.CommonDeleteMapper;
30  import org.mybatis.dynamic.sql.util.mybatis3.CommonInsertMapper;
31  import org.mybatis.dynamic.sql.util.mybatis3.CommonUpdateMapper;

34  public interface AnimalDataMapper extends CommonDeleteMapper, CommonInsertMapper<AnimalData>, CommonUpdateMapper {

37      @Results(id="AnimalDataResult", value={
38          @Result(column="id", property="id", id=true),
39          @Result(column="animal_name", property="animalName"),
40          @Result(column="brain_weight", property="brainWeight"),
41          @Result(column="body_weight", property="bodyWeight")
42      })
43      List<AnimalData> selectMany(SelectStatementProvider selectStatement);
44  
45      @SelectProvider(type = SqlProviderAdapter.class, method = "select")
46      @ResultMap("AnimalDataResult")
47      List<AnimalData> selectManyWithRowBounds(SelectStatementProvider selectStatement, RowBounds rowBounds);
48  
49      @SelectProvider(type = SqlProviderAdapter.class, method = "select")
50      @ResultMap("AnimalDataResult")

52  
53      @Select({
54          "select id, animal_name, brain_weight, body_weight",
55          "from AnimalData",
56          "${whereClause}"
57      })
58      @ResultMap("AnimalDataResult")
59      List<AnimalData> selectWithWhereClause(WhereClauseProvider whereClause);
60  
61      @Select({
62          "select a.id, a.animal_name, a.brain_weight, a.body_weight",
63          "from AnimalData a",
64          "${whereClause}"
65      })
66      @ResultMap("AnimalDataResult")
67      List<AnimalData> selectWithWhereClauseAndAlias(WhereClauseProvider whereClause);
68  
69      @Select({
70          "select id, animal_name, brain_weight, body_weight",
71          "from AnimalData",
72          "${whereClauseProvider.whereClause}",
73          "order by id",
74          "OFFSET #{offset,jdbcType=INTEGER} LIMIT #{limit,jdbcType=INTEGER}"
75      })
76      @ResultMap("AnimalDataResult")
77      List<AnimalData> selectWithWhereClauseLimitAndOffset(@Param("whereClauseProvider") WhereClauseProvider whereClause,
78              @Param("limit") int limit, @Param("offset") int offset);
79  
80      @Select({
81          "select b.id, b.animal_name, b.brain_weight, b.body_weight",
82          "from AnimalData b",
83          "${whereClauseProvider.whereClause}",
84          "order by id",
85          "OFFSET #{offset,jdbcType=INTEGER} LIMIT #{limit,jdbcType=INTEGER}"
86      })
87      @ResultMap("AnimalDataResult")
88      List<AnimalData> selectWithWhereClauseAliasLimitAndOffset(@Param("whereClauseProvider") WhereClauseProvider whereClause,
89                                                                @Param("limit") int limit, @Param("offset") int offset);
90  }
