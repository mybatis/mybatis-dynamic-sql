18  import java.util.List;
19  
20  import org.apache.ibatis.annotations.Param;
21  import org.apache.ibatis.annotations.Result;
22  import org.apache.ibatis.annotations.ResultMap;
23  import org.apache.ibatis.annotations.Results;
24  import org.apache.ibatis.annotations.Select;
25  import org.apache.ibatis.annotations.SelectProvider;
26  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
27  import org.mybatis.dynamic.sql.util.SqlProviderAdapter;
28  import org.mybatis.dynamic.sql.util.mybatis3.CommonDeleteMapper;
29  import org.mybatis.dynamic.sql.util.mybatis3.CommonInsertMapper;
30  import org.mybatis.dynamic.sql.util.mybatis3.CommonUpdateMapper;
31  import org.mybatis.dynamic.sql.where.render.WhereClauseProvider;
32  
33  public interface AnimalDataMapper extends CommonDeleteMapper, CommonInsertMapper<AnimalData>, CommonUpdateMapper {
34  
35      @SelectProvider(type=SqlProviderAdapter.class, method="select")
36      @Results(id="AnimalDataResult", value={
37          @Result(column="id", property="id", id=true),
38          @Result(column="animal_name", property="animalName"),
39          @Result(column="brain_weight", property="brainWeight"),
40          @Result(column="body_weight", property="bodyWeight")
41      })
42      List<AnimalData> selectMany(SelectStatementProvider selectStatement);
43  
44      @SelectProvider(type=SqlProviderAdapter.class, method="select")
45      @ResultMap("AnimalDataResult")
46      AnimalData selectOne(SelectStatementProvider selectStatement);
47  
48      @Select({
49          "select id, animal_name, brain_weight, body_weight",
50          "from AnimalData",
51          "${whereClause}"
52      })
53      @ResultMap("AnimalDataResult")
54      List<AnimalData> selectWithWhereClause(WhereClauseProvider whereClause);
55  
56      @Select({
57          "select a.id, a.animal_name, a.brain_weight, a.body_weight",
58          "from AnimalData a",
59          "${whereClause}"
60      })
61      @ResultMap("AnimalDataResult")
62      List<AnimalData> selectWithWhereClauseAndAlias(WhereClauseProvider whereClause);
63  
64      @Select({
65          "select id, animal_name, brain_weight, body_weight",
66          "from AnimalData",
67          "${whereClauseProvider.whereClause}",
68          "order by id",
69          "OFFSET #{offset,jdbcType=INTEGER} LIMIT #{limit,jdbcType=INTEGER}"
70      })
71      @ResultMap("AnimalDataResult")
72      List<AnimalData> selectWithWhereClauseLimitAndOffset(@Param("whereClauseProvider") WhereClauseProvider whereClause,
73                                                           @Param("limit") int limit, @Param("offset") int offset);
74  
75      @Select({
76          "select b.id, b.animal_name, b.brain_weight, b.body_weight",
77          "from AnimalData b",
78          "${whereClauseProvider.whereClause}",
79          "order by id",
80          "OFFSET #{offset,jdbcType=INTEGER} LIMIT #{limit,jdbcType=INTEGER}"
81      })
82      @ResultMap("AnimalDataResult")
83      List<AnimalData> selectWithWhereClauseAliasLimitAndOffset(@Param("whereClauseProvider") WhereClauseProvider whereClause,
84                                                                @Param("limit") int limit, @Param("offset") int offset);
85  }
