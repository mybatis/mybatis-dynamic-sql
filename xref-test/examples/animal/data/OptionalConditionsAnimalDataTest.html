69      void testAllIgnored() {
70          try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
71              AnimalDataMapper mapper = sqlSession.getMapper(AnimalDataMapper.class);
72              SelectStatementProvider selectStatement = select(id, animalName, bodyWeight, brainWeight)
73                      .from(animalData)
74                      .where(id, isGreaterThanWhenPresent(NULL_INTEGER))  // the where clause should not render
75                      .orderBy(id)
76                      .configureStatement(c -> c.setNonRenderingWhereClauseAllowed(true))
77                      .build()
78                      .render(RenderingStrategies.MYBATIS3);
79              List<AnimalData> animals = mapper.selectMany(selectStatement);
80              assertAll(
81                      () -> assertThat(selectStatement.getSelectStatement()).isEqualTo("select id, animal_name, body_weight, brain_weight from AnimalData order by id"),
82                      () -> assertThat(animals).hasSize(65),
83                      () -> assertThat(animals.get(0).getId()).isEqualTo(1),
84                      () -> assertThat(animals.get(1).getId()).isEqualTo(2)
85              );
86          }
