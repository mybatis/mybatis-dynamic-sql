16  package org.mybatis.dynamic.sql.util.spring;
17  
18  import java.util.List;
19  import java.util.Objects;
20  import java.util.Optional;
21  
22  import org.mybatis.dynamic.sql.delete.DeleteModel;
23  import org.mybatis.dynamic.sql.delete.render.DeleteStatementProvider;
24  import org.mybatis.dynamic.sql.insert.BatchInsertModel;
25  import org.mybatis.dynamic.sql.insert.GeneralInsertModel;
26  import org.mybatis.dynamic.sql.insert.InsertModel;
27  import org.mybatis.dynamic.sql.insert.MultiRowInsertModel;
28  import org.mybatis.dynamic.sql.insert.render.BatchInsert;
29  import org.mybatis.dynamic.sql.insert.render.GeneralInsertStatementProvider;
30  import org.mybatis.dynamic.sql.insert.render.InsertStatementProvider;
31  import org.mybatis.dynamic.sql.insert.render.MultiRowInsertStatementProvider;
32  import org.mybatis.dynamic.sql.render.RenderingStrategies;
33  import org.mybatis.dynamic.sql.select.SelectModel;
34  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
35  import org.mybatis.dynamic.sql.update.UpdateModel;
36  import org.mybatis.dynamic.sql.update.render.UpdateStatementProvider;
37  import org.mybatis.dynamic.sql.util.Buildable;
38  import org.springframework.dao.EmptyResultDataAccessException;
39  import org.springframework.jdbc.core.RowMapper;
40  import org.springframework.jdbc.core.namedparam.BeanPropertySqlParameterSource;
41  import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
42  import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;
43  import org.springframework.jdbc.support.KeyHolder;
44  
45  public class NamedParameterJdbcTemplateExtensions {
46      private final NamedParameterJdbcTemplate template;
47  
48      public NamedParameterJdbcTemplateExtensions(NamedParameterJdbcTemplate template) {
49          this.template = Objects.requireNonNull(template);
50      }
51  
52      public long count(Buildable<SelectModel> countStatement) {
53          return count(countStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER));
54      }
55  
56      public long count(SelectStatementProvider countStatement) {
57          return template.queryForObject(countStatement.getSelectStatement(), countStatement.getParameters(), Long.class);
58      }
59  
60      public int delete(Buildable<DeleteModel> deleteStatement) {
61          return delete(deleteStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER));
62      }
63  
64      public int delete(DeleteStatementProvider deleteStatement) {
65          return template.update(deleteStatement.getDeleteStatement(), deleteStatement.getParameters());
66      }
67  
68      public int generalInsert(Buildable<GeneralInsertModel> insertStatement) {
69          return generalInsert(insertStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER));
70      }
71  
72      public int generalInsert(GeneralInsertStatementProvider insertStatement) {
73          return template.update(insertStatement.getInsertStatement(), insertStatement.getParameters());
74      }
75  
76      public int generalInsert(Buildable<GeneralInsertModel> insertStatement, KeyHolder keyHolder) {
77          return generalInsert(insertStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER), keyHolder);
78      }
79  
80      public int generalInsert(GeneralInsertStatementProvider insertStatement, KeyHolder keyHolder) {
81          return template.update(insertStatement.getInsertStatement(),
82                  new MapSqlParameterSource(insertStatement.getParameters()), keyHolder);
83      }
84  
85      public <T> int insert(Buildable<InsertModel<T>> insertStatement) {
86          return insert(insertStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER));
87      }
88  
89      public <T> int insert(InsertStatementProvider<T> insertStatement) {
90          return template.update(insertStatement.getInsertStatement(),
91                  new BeanPropertySqlParameterSource(insertStatement));
92      }
93  
94      public <T> int insert(Buildable<InsertModel<T>> insertStatement, KeyHolder keyHolder) {
95          return insert(insertStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER), keyHolder);
96      }
97  
98      public <T> int insert(InsertStatementProvider<T> insertStatement, KeyHolder keyHolder) {
99          return template.update(insertStatement.getInsertStatement(),
100                 new BeanPropertySqlParameterSource(insertStatement), keyHolder);
101     }
102 
103     public <T> int[] insertBatch(Buildable<BatchInsertModel<T>> insertStatement) {
104         return insertBatch(insertStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER));
105     }
106 
107     public <T> int[] insertBatch(BatchInsert<T> insertStatement) {
108         return template.batchUpdate(insertStatement.getInsertStatementSQL(),
109                 BatchInsertUtility.createBatch(insertStatement.getRecords()));
110     }
111 
112     public <T> int insertMultiple(Buildable<MultiRowInsertModel<T>> insertStatement) {
113         return insertMultiple(insertStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER));
114     }
115 
116     public <T> int insertMultiple(MultiRowInsertStatementProvider<T> insertStatement) {
117         return template.update(insertStatement.getInsertStatement(),
118                 new BeanPropertySqlParameterSource(insertStatement));
119     }
120 
121     public <T> int insertMultiple(Buildable<MultiRowInsertModel<T>> insertStatement, KeyHolder keyHolder) {
122         return insertMultiple(insertStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER), keyHolder);
123     }
124 
125     public <T> int insertMultiple(MultiRowInsertStatementProvider<T> insertStatement, KeyHolder keyHolder) {
126         return template.update(insertStatement.getInsertStatement(),
127                 new BeanPropertySqlParameterSource(insertStatement), keyHolder);
128     }
129 
130     public <T> List<T> selectList(Buildable<SelectModel> selectStatement, RowMapper<T> rowMapper) {
131         return selectList(selectStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER), rowMapper);
132     }
133 
134     public <T> List<T> selectList(SelectStatementProvider selectStatement, RowMapper<T> rowMapper) {
135         return template.query(selectStatement.getSelectStatement(), selectStatement.getParameters(), rowMapper);
136     }
137 
138     public <T> Optional<T> selectOne(Buildable<SelectModel> selectStatement, RowMapper<T> rowMapper) {
139         return selectOne(selectStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER), rowMapper);
140     }
141 
142     public <T> Optional<T> selectOne(SelectStatementProvider selectStatement, RowMapper<T> rowMapper) {
143         T result;
144         try {
145             result = template.queryForObject(selectStatement.getSelectStatement(), selectStatement.getParameters(),
146                             rowMapper);
147         } catch (EmptyResultDataAccessException e) {
148             result = null;
149         }
150 
151         return Optional.ofNullable(result);
152     }
153 
154     public int update(Buildable<UpdateModel> updateStatement) {
155         return update(updateStatement.build().render(RenderingStrategies.SPRING_NAMED_PARAMETER));
156     }
157 
158     public int update(UpdateStatementProvider updateStatement) {
159         return template.update(updateStatement.getUpdateStatement(), updateStatement.getParameters());
160     }
161 }
