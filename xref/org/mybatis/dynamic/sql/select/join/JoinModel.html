16  package org.mybatis.dynamic.sql.select.join;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  import java.util.Objects;
21  import java.util.function.Function;
22  import java.util.stream.Stream;
23  
24  import org.mybatis.dynamic.sql.TableExpression;
25  import org.mybatis.dynamic.sql.exception.InvalidSqlException;
26  import org.mybatis.dynamic.sql.util.Messages;
27  
28  public class JoinModel {
29      private final List<JoinSpecification> joinSpecifications = new ArrayList<>();
30  
31      private JoinModel(List<JoinSpecification> joinSpecifications) {
32          Objects.requireNonNull(joinSpecifications);
33          if (joinSpecifications.isEmpty()) {
34              throw new InvalidSqlException(Messages.getString("ERROR.15")); //$NON-NLS-1$
35          }
36  
37          this.joinSpecifications.addAll(joinSpecifications);
38      }
39  
40      public <R> Stream<R> mapJoinSpecifications(Function<JoinSpecification, R> mapper) {
41          return joinSpecifications.stream().map(mapper);
42      }
43  
44      public static JoinModel of(List<JoinSpecification> joinSpecifications) {
45          return new JoinModel(joinSpecifications);
46      }
47  
48      public boolean containsSubQueries() {
49          return joinSpecifications.stream()
50                  .map(JoinSpecification::table)
51                  .anyMatch(TableExpression::isSubQuery);
52      }
53  }
