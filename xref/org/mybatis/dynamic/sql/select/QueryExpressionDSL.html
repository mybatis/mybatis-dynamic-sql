40  public class QueryExpressionDSL<R> extends AbstractQueryExpressionDSL<QueryExpressionDSL<R>, R>
41          implements Buildable<R> {
42  
43      private final String connector;
44      private final SelectDSL<R> selectDSL;
45      private final boolean isDistinct;
46      private final List<BasicColumn> selectList;
47      private final QueryExpressionWhereBuilder whereBuilder = new QueryExpressionWhereBuilder();
48      private GroupByModel groupByModel;
49      
50      QueryExpressionDSL(FromGatherer<R> fromGatherer) {
51          super(fromGatherer.table);
52          connector = fromGatherer.connector;
53          selectList = fromGatherer.selectList;
54          isDistinct = fromGatherer.isDistinct;
55          selectDSL = Objects.requireNonNull(fromGatherer.selectDSL);
56      }
57      
58      QueryExpressionDSL(FromGatherer<R> fromGatherer, String tableAlias) {
59          this(fromGatherer);
60          tableAliases.put(fromGatherer.table, tableAlias);
61      }
62      
63      public QueryExpressionWhereBuilder where() {
64          return whereBuilder;
65      }
66  
67      public <T> QueryExpressionWhereBuilder where(BindableColumn<T> column, VisitableCondition<T> condition,
68              SqlCriterion<?>...subCriteria) {
69          whereBuilder.where(column, condition, subCriteria);
70          return whereBuilder;
71      }
72  
73      public QueryExpressionWhereBuilder applyWhere(WhereApplier whereApplier) {
74          return whereBuilder.applyWhere(whereApplier);
75      }
76  
77      @NotNull
78      @Override
79      public R build() {
80          return selectDSL.build();
81      }

145     protected QueryExpressionModel buildModel() {
146         return QueryExpressionModel.withSelectList(selectList)
147                 .withConnector(connector)
148                 .withTable(table())
149                 .isDistinct(isDistinct)
150                 .withTableAliases(tableAliases)
151                 .withWhereModel(whereBuilder.buildWhereModel())
152                 .withJoinModel(buildJoinModel().orElse(null))
153                 .withGroupByModel(groupByModel)
154                 .build();
155     }

174     public static class FromGatherer<R> {
175         private final String connector;
176         private final List<BasicColumn> selectList;
177         private final SelectDSL<R> selectDSL;
178         private final boolean isDistinct;
179         private SqlTable table;
180         
181         public FromGatherer(Builder<R> builder) {
182             this.connector = builder.connector;
183             this.selectList = Objects.requireNonNull(builder.selectList);
184             this.selectDSL = Objects.requireNonNull(builder.selectDSL);
185             this.isDistinct = builder.isDistinct;
186         }
187         
188         public QueryExpressionDSL<R> from(SqlTable table) {
189             this.table = table;
190             return selectDSL.newQueryExpression(this);
191         }
192 
193         public QueryExpressionDSL<R> from(SqlTable table, String tableAlias) {
194             this.table = table;
195             return selectDSL.newQueryExpression(this, tableAlias);
196         }

299     public class JoinSpecificationFinisher implements Buildable<R> {
300         private final JoinSpecification.Builder joinSpecificationBuilder;
301         
302         public JoinSpecificationFinisher(SqlTable table, BasicColumn joinColumn,
303                 JoinCondition joinCondition, JoinType joinType) {
304             JoinCriterion joinCriterion = new JoinCriterion.Builder()
305                     .withConnector("on") //$NON-NLS-1$
306                     .withJoinColumn(joinColumn)
307                     .withJoinCondition(joinCondition)
308                     .build();
309 
310             joinSpecificationBuilder = JoinSpecification.withJoinTable(table)
311                     .withJoinType(joinType)
312                     .withJoinCriterion(joinCriterion);
313 
314             addJoinSpecificationBuilder(joinSpecificationBuilder);
315         }
316 
317         public JoinSpecificationFinisher(SqlTable table, BasicColumn joinColumn,
318                 JoinCondition joinCondition, JoinType joinType, JoinCriterion...andJoinCriteria) {
319             JoinCriterion onJoinCriterion = new JoinCriterion.Builder()
320                     .withConnector("on") //$NON-NLS-1$
321                     .withJoinColumn(joinColumn)
322                     .withJoinCondition(joinCondition)
323                     .build();
324             
325             joinSpecificationBuilder = JoinSpecification.withJoinTable(table)
326                     .withJoinType(joinType)
327                     .withJoinCriterion(onJoinCriterion)
328                     .withJoinCriteria(Arrays.asList(andJoinCriteria));
329 
330             addJoinSpecificationBuilder(joinSpecificationBuilder);
331         }
