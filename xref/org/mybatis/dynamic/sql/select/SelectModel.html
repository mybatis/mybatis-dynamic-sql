16  package org.mybatis.dynamic.sql.select;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  import java.util.Objects;
21  import java.util.stream.Stream;
22  
23  import org.jetbrains.annotations.NotNull;
24  import org.mybatis.dynamic.sql.render.RenderingContext;
25  import org.mybatis.dynamic.sql.render.RenderingStrategy;
26  import org.mybatis.dynamic.sql.select.render.SelectRenderer;
27  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
28  import org.mybatis.dynamic.sql.util.Validator;
29  
30  public class SelectModel extends AbstractSelectModel {
31      private final List<QueryExpressionModel> queryExpressions;
32  
33      private SelectModel(Builder builder) {
34          super(builder);
35          queryExpressions = Objects.requireNonNull(builder.queryExpressions);
36          Validator.assertNotEmpty(queryExpressions, "ERROR.14"); //$NON-NLS-1$
37      }
38  
39      public Stream<QueryExpressionModel> queryExpressions() {
40          return queryExpressions.stream();
41      }
42  
43      @NotNull
44      public SelectStatementProvider render(RenderingStrategy renderingStrategy) {
45          RenderingContext renderingContext = RenderingContext.withRenderingStrategy(renderingStrategy)
46                  .withStatementConfiguration(statementConfiguration)
47                  .build();
48          return render(renderingContext);
49      }
50  
51      /**
52       * This version is for rendering sub-queries, union queries, etc.
53       *
54       * @param renderingContext the rendering context
55       * @return a rendered select statement and parameters
56       */
57      @NotNull
58      public SelectStatementProvider render(RenderingContext renderingContext) {
59          return SelectRenderer.withSelectModel(this)
60                  .withRenderingContext(renderingContext)
61                  .build()
62                  .render();
63      }
64  
65      public static Builder withQueryExpressions(List<QueryExpressionModel> queryExpressions) {
66          return new Builder().withQueryExpressions(queryExpressions);
67      }
68  
69      public static class Builder extends AbstractBuilder<Builder> {
70          private final List<QueryExpressionModel> queryExpressions = new ArrayList<>();
71  
72          public Builder withQueryExpression(QueryExpressionModel queryExpression) {
73              this.queryExpressions.add(queryExpression);
74              return this;
75          }
76  
77          public Builder withQueryExpressions(List<QueryExpressionModel> queryExpressions) {
78              this.queryExpressions.addAll(queryExpressions);
79              return this;
80          }
81  
82          @Override
83          protected Builder getThis() {
84              return this;
85          }
86  
87          public SelectModel build() {
88              return new SelectModel(this);
89          }
90      }
91  }
