16  package org.mybatis.dynamic.sql.select.render;
17  
18  import java.util.concurrent.atomic.AtomicInteger;
19  
20  import org.mybatis.dynamic.sql.exception.InvalidSqlException;
21  import org.mybatis.dynamic.sql.render.RenderingStrategy;
22  import org.mybatis.dynamic.sql.select.PagingModel;
23  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
24  import org.mybatis.dynamic.sql.util.InternalError;
25  import org.mybatis.dynamic.sql.util.Messages;
26  
27  public class FetchFirstPagingModelRenderer {
28      private final RenderingStrategy renderingStrategy;
29      private final PagingModel pagingModel;
30      private final AtomicInteger sequence;
31  
32      public FetchFirstPagingModelRenderer(RenderingStrategy renderingStrategy,
33              PagingModel pagingModel, AtomicInteger sequence) {
34          this.renderingStrategy = renderingStrategy;
35          this.pagingModel = pagingModel;
36          this.sequence = sequence;
37      }
38  
39      public FragmentAndParameters render() {
40          return pagingModel.offset()
41                  .map(this::renderWithOffset)
42                  .orElseGet(this::renderFetchFirstRowsOnly);
43      }
44  
45      private FragmentAndParameters renderWithOffset(Long offset) {
46          return pagingModel.fetchFirstRows()
47                  .map(ffr -> renderOffsetAndFetchFirstRows(offset, ffr))
48                  .orElseGet(() -> renderOffsetOnly(offset));
49      }
50  
51      private FragmentAndParameters renderFetchFirstRowsOnly() {
52          return pagingModel.fetchFirstRows().map(this::renderFetchFirstRowsOnly)
53                  .orElseThrow(() ->
54                          new InvalidSqlException(Messages.getInternalErrorString(InternalError.INTERNAL_ERROR_13)));
55      }
56  
57      private FragmentAndParameters renderFetchFirstRowsOnly(Long fetchFirstRows) {
58          String mapKey = renderingStrategy.formatParameterMapKey(sequence);
59          return FragmentAndParameters
60                  .withFragment("fetch first " + renderPlaceholder(mapKey) //$NON-NLS-1$
61                      + " rows only") //$NON-NLS-1$
62                  .withParameter(mapKey, fetchFirstRows)
63                  .build();
64      }
65  
66      private FragmentAndParameters renderOffsetOnly(Long offset) {
67          String mapKey = renderingStrategy.formatParameterMapKey(sequence);
68          return FragmentAndParameters.withFragment("offset " + renderPlaceholder(mapKey) //$NON-NLS-1$
69                  + " rows") //$NON-NLS-1$
70                  .withParameter(mapKey, offset)
71                  .build();
72      }
73  
74      private FragmentAndParameters renderOffsetAndFetchFirstRows(Long offset, Long fetchFirstRows) {
75          String mapKey1 = renderingStrategy.formatParameterMapKey(sequence);
76          String mapKey2 = renderingStrategy.formatParameterMapKey(sequence);
77          return FragmentAndParameters.withFragment("offset " + renderPlaceholder(mapKey1) //$NON-NLS-1$
78                  + " rows fetch first " + renderPlaceholder(mapKey2) //$NON-NLS-1$
79                  + " rows only") //$NON-NLS-1$
80                  .withParameter(mapKey1, offset)
81                  .withParameter(mapKey2, fetchFirstRows)
82                  .build();
83      }
84  
85      private String renderPlaceholder(String parameterName) {
86          return renderingStrategy.getFormattedJdbcPlaceholder(RenderingStrategy.DEFAULT_PARAMETER_PREFIX,
87                  parameterName);
88      }
89  }
