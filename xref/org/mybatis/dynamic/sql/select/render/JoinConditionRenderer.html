16  package org.mybatis.dynamic.sql.select.render;
17  
18  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceBefore;
19  
20  import java.util.Objects;
21  
22  import org.mybatis.dynamic.sql.BindableColumn;
23  import org.mybatis.dynamic.sql.render.RenderedParameterInfo;
24  import org.mybatis.dynamic.sql.render.RenderingContext;
25  import org.mybatis.dynamic.sql.select.join.ColumnBasedJoinCondition;
26  import org.mybatis.dynamic.sql.select.join.JoinConditionVisitor;
27  import org.mybatis.dynamic.sql.select.join.TypedJoinCondition;
28  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
29  
30  public class JoinConditionRenderer<T> implements JoinConditionVisitor<T, FragmentAndParameters> {
31      private final BindableColumn<T> leftColumn;
32      private final RenderingContext renderingContext;
33  
34      private JoinConditionRenderer(Builder<T> builder) {
35          leftColumn = Objects.requireNonNull(builder.leftColumn);
36          renderingContext = Objects.requireNonNull(builder.renderingContext);
37      }
38  
39      @Override
40      public FragmentAndParameters visit(TypedJoinCondition<T> condition) {
41          RenderedParameterInfo parameterInfo = renderingContext.calculateParameterInfo(leftColumn);
42  
43          return FragmentAndParameters
44                  .withFragment(condition.operator() + spaceBefore(parameterInfo.renderedPlaceHolder()))
45                  .withParameter(parameterInfo.parameterMapKey(), condition.value())
46                  .build();
47      }
48  
49      @Override
50      public FragmentAndParameters visit(ColumnBasedJoinCondition<T> condition) {
51          return condition.rightColumn().render(renderingContext)
52                  .mapFragment(s -> condition.operator() + spaceBefore(s));
53      }
54  
55      public static class Builder<T> {
56          private BindableColumn<T> leftColumn;
57          private RenderingContext renderingContext;
58  
59          public Builder<T> withLeftColumn(BindableColumn<T> leftColumn) {
60              this.leftColumn = leftColumn;
61              return this;
62          }
63  
64          public Builder<T> withRenderingContext(RenderingContext renderingContext) {
65              this.renderingContext = renderingContext;
66              return this;
67          }
68  
69          public JoinConditionRenderer<T> build() {
70              return new JoinConditionRenderer<>(this);
71          }
72      }
73  }
