16  package org.mybatis.dynamic.sql.select.render;
17  
18  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceAfter;
19  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceBefore;
20  
21  import java.util.Collections;
22  import java.util.Map;
23  import java.util.Objects;
24  import java.util.Optional;
25  import java.util.concurrent.atomic.AtomicInteger;
26  import java.util.stream.Collectors;
27  
28  import org.mybatis.dynamic.sql.BasicColumn;
29  import org.mybatis.dynamic.sql.SqlTable;
30  import org.mybatis.dynamic.sql.render.RenderingStrategy;
31  import org.mybatis.dynamic.sql.select.GroupByModel;
32  import org.mybatis.dynamic.sql.select.QueryExpressionModel;
33  import org.mybatis.dynamic.sql.select.join.JoinModel;
34  import org.mybatis.dynamic.sql.util.CustomCollectors;
35  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
36  import org.mybatis.dynamic.sql.where.WhereModel;
37  import org.mybatis.dynamic.sql.where.render.WhereClauseProvider;
38  import org.mybatis.dynamic.sql.where.render.WhereRenderer;
39  
40  public class QueryExpressionRenderer {
41      private QueryExpressionModel queryExpression;
42      private RenderingStrategy renderingStrategy;
43      private AtomicInteger sequence;
44      
45      private QueryExpressionRenderer(Builder builder) {
46          queryExpression = Objects.requireNonNull(builder.queryExpression);
47          renderingStrategy = Objects.requireNonNull(builder.renderingStrategy);
48          sequence = Objects.requireNonNull(builder.sequence);
49      }
50      
51      public FragmentAndParameters render() {
52          Optional<WhereClauseProvider> whereClause = queryExpression.whereModel().flatMap(this::renderWhereClause);
53          
54          return FragmentAndParameters.withFragment(calculateQueryExpression(whereClause))
55                  .withParameters(calculateParameters(whereClause))
56                  .build();
57      }
58      
59      private String calculateQueryExpression(Optional<WhereClauseProvider> whereClause) {
60          return spaceAfter(queryExpression.connector())
61                  + "select " //$NON-NLS-1$
62                  + (queryExpression.isDistinct() ? "distinct " : "") //$NON-NLS-1$ //$NON-NLS-2$
63                  + calculateColumnList()
64                  + " from " //$NON-NLS-1$
65                  + calculateTableName(queryExpression.table())
66                  + spaceBefore(queryExpression.joinModel().map(this::renderJoin))
67                  + spaceBefore(whereClause.map(WhereClauseProvider::getWhereClause))
68                  + spaceBefore(queryExpression.groupByModel().map(this::renderGroupBy));
69      }
70  
71      private Map<String, Object> calculateParameters(Optional<WhereClauseProvider> whereClause) {
72          return whereClause
73                  .map(WhereClauseProvider::getParameters)
74                  .orElse(Collections.emptyMap());
75      }
76  
77      private String calculateColumnList() {
78          return queryExpression.mapColumns(this::applyTableAndColumnAlias)
79                  .collect(Collectors.joining(", ")); //$NON-NLS-1$
80      }
81  
82      private String calculateTableName(SqlTable table) {
83          return queryExpression.calculateTableNameIncludingAlias(table);
84      }
85      
86      private String applyTableAndColumnAlias(BasicColumn selectListItem) {
87          return selectListItem.renderWithTableAndColumnAlias(queryExpression.tableAliasCalculator());
88      }
89      
90      private String renderJoin(JoinModel joinModel) {
91          return JoinRenderer.withJoinModel(joinModel)
92                  .withQueryExpression(queryExpression)
93                  .build()
94                  .render();
95      }
96      
97      private Optional<WhereClauseProvider> renderWhereClause(WhereModel whereModel) {
98          return WhereRenderer.withWhereModel(whereModel)
99                  .withRenderingStrategy(renderingStrategy)
100                 .withTableAliasCalculator(queryExpression.tableAliasCalculator())
101                 .withSequence(sequence)
102                 .build()
103                 .render();
104     }
105 
106     private String renderGroupBy(GroupByModel groupByModel) {
107         return groupByModel.mapColumns(this::applyTableAlias)
108                 .collect(CustomCollectors.joining(", ", "group by ", "")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
109     }
110     
111     private String applyTableAlias(BasicColumn column) {
112         return column.renderWithTableAlias(queryExpression.tableAliasCalculator());
113     }
114     
115     public static Builder withQueryExpression(QueryExpressionModel model) {
116         return new Builder().withQueryExpression(model);
117     }
118     
119     public static class Builder {
120         private QueryExpressionModel queryExpression;
121         private RenderingStrategy renderingStrategy;
122         private AtomicInteger sequence;
123         
124         public Builder withQueryExpression(QueryExpressionModel queryExpression) {
125             this.queryExpression = queryExpression;
126             return this;
127         }
128         
129         public Builder withRenderingStrategy(RenderingStrategy renderingStrategy) {
130             this.renderingStrategy = renderingStrategy;
131             return this;
132         }
133         
134         public Builder withSequence(AtomicInteger sequence) {
135             this.sequence = sequence;
136             return this;
137         }
138         
139         public QueryExpressionRenderer build() {
140             return new QueryExpressionRenderer(this);
141         }
142     }
143 }
