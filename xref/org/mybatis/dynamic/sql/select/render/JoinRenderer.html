16  package org.mybatis.dynamic.sql.select.render;
17  
18  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceAfter;
19  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceBefore;
20  
21  import java.util.Objects;
22  import java.util.stream.Collectors;
23  
24  import org.mybatis.dynamic.sql.BasicColumn;
25  import org.mybatis.dynamic.sql.select.QueryExpressionModel;
26  import org.mybatis.dynamic.sql.select.join.JoinCriterion;
27  import org.mybatis.dynamic.sql.select.join.JoinModel;
28  import org.mybatis.dynamic.sql.select.join.JoinSpecification;
29  
30  public class JoinRenderer {
31      private JoinModel joinModel;
32      private QueryExpressionModel queryExpression;
33      
34      private JoinRenderer(Builder builder) {
35          joinModel = Objects.requireNonNull(builder.joinModel);
36          queryExpression = Objects.requireNonNull(builder.queryExpression);
37      }
38      
39      public String render() {
40          return joinModel.mapJoinSpecifications(this::toRenderedString)
41                  .collect(Collectors.joining(" ")); //$NON-NLS-1$
42      }
43      
44      private String toRenderedString(JoinSpecification joinSpecification) {
45          return spaceAfter(joinSpecification.joinType().shortType())
46                  + "join" //$NON-NLS-1$
47                  + spaceBefore(queryExpression.calculateTableNameIncludingAlias(joinSpecification.table()))
48                  + spaceBefore(renderConditions(joinSpecification));
49      }
50      
51      private String renderConditions(JoinSpecification joinSpecification) {
52          return joinSpecification.mapJoinCriteria(this::renderCriterion)
53                  .collect(Collectors.joining(" ")); //$NON-NLS-1$
54      }
55      
56      private String renderCriterion(JoinCriterion joinCriterion) {
57          return joinCriterion.connector()
58                  + spaceBefore(applyTableAlias(joinCriterion.leftColumn()))
59                  + spaceBefore(joinCriterion.operator())
60                  + spaceBefore(applyTableAlias(joinCriterion.rightColumn()));
61      }
62      
63      private String applyTableAlias(BasicColumn column) {
64          return column.renderWithTableAlias(queryExpression.tableAliasCalculator());
65      }
66      
67      public static Builder withJoinModel(JoinModel joinModel) {
68          return new Builder().withJoinModel(joinModel);
69      }
70      
71      public static class Builder {
72          private JoinModel joinModel;
73          private QueryExpressionModel queryExpression;
74          
75          public Builder withJoinModel(JoinModel joinModel) {
76              this.joinModel = joinModel;
77              return this;
78          }
79          
80          public Builder withQueryExpression(QueryExpressionModel queryExpression) {
81              this.queryExpression = queryExpression;
82              return this;
83          }
84          
85          public JoinRenderer build() {
86              return new JoinRenderer(this);
87          }
88      }
89  }
