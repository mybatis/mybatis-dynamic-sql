16  package org.mybatis.dynamic.sql.render;
17  
18  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceBefore;
19  
20  import java.util.Objects;
21  import java.util.concurrent.atomic.AtomicInteger;
22  
23  import org.mybatis.dynamic.sql.BindableColumn;
24  import org.mybatis.dynamic.sql.SqlColumn;
25  import org.mybatis.dynamic.sql.SqlTable;
26  
27  /**
28   * This class encapsulates all the supporting items related to rendering, and contains many utility methods
29   * used during the rendering process.
30   *
31   * @since 1.5.1
32   * @author Jeff Butler
33   */
34  public class RenderingContext {
35  
36      private final RenderingStrategy renderingStrategy;
37      private final AtomicInteger sequence;
38      private final TableAliasCalculator tableAliasCalculator;
39      private final String configuredParameterName;
40      private final String calculatedParameterName;
41  
42      private RenderingContext(Builder builder) {
43          renderingStrategy = Objects.requireNonNull(builder.renderingStrategy);
44          configuredParameterName = builder.parameterName;
45          tableAliasCalculator = Objects.requireNonNull(builder.tableAliasCalculator);
46  
47          // reasonable defaults
48          sequence = builder.sequence == null ? new AtomicInteger(1) : builder.sequence;
49          calculatedParameterName = builder.parameterName == null ? RenderingStrategy.DEFAULT_PARAMETER_PREFIX
50                  : builder.parameterName + "." + RenderingStrategy.DEFAULT_PARAMETER_PREFIX;  //$NON-NLS-1$
51      }
52  
53      public TableAliasCalculator tableAliasCalculator() {
54          // this method can be removed when the renderWithTableAlias method is removed from BasicColumn
55          return tableAliasCalculator;
56      }
57  
58      private String nextMapKey() {
59          return renderingStrategy.formatParameterMapKey(sequence);
60      }
61  
62      private String renderedPlaceHolder(String mapKey) {
63          return renderingStrategy.getFormattedJdbcPlaceholder(calculatedParameterName, mapKey);
64      }
65  
66      private <T> String renderedPlaceHolder(String mapKey, BindableColumn<T> column) {
67          return  column.renderingStrategy().orElse(renderingStrategy)
68                  .getFormattedJdbcPlaceholder(column, calculatedParameterName, mapKey);
69      }
70  
71      public RenderedParameterInfo calculateParameterInfo() {
72          String mapKey = nextMapKey();
73          return new RenderedParameterInfo(mapKey, renderedPlaceHolder(mapKey));
74      }
75  
76      public <T> RenderedParameterInfo calculateParameterInfo(BindableColumn<T> column) {
77          String mapKey = nextMapKey();
78          return new RenderedParameterInfo(mapKey, renderedPlaceHolder(mapKey, column));
79      }
80  
81      public <T> String aliasedColumnName(SqlColumn<T> column) {
82          return tableAliasCalculator.aliasForColumn(column.table())
83                  .map(alias -> aliasedColumnName(column, alias))
84                  .orElseGet(column::name);
85      }
86  
87      public <T> String aliasedColumnName(SqlColumn<T> column, String explicitAlias) {
88          return explicitAlias + "." + column.name();  //$NON-NLS-1$
89      }
90  
91      public String aliasedTableName(SqlTable table) {
92          return tableAliasCalculator.aliasForTable(table)
93                  .map(a -> table.tableNameAtRuntime() + spaceBefore(a))
94                  .orElseGet(table::tableNameAtRuntime);
95      }
96  
97      /**
98       * Create a new rendering context based on this, with the table alias calculator modified to include the
99       * specified child table alias calculator. This is used by the query expression renderer when the alias calculator
100      * may change during rendering.
101      *
102      * @param childTableAliasCalculator the child table alias calculator
103      * @return a new rendering context whose table alias calculator is composed of the former calculator as parent, and
104      *     the new child calculator
105      */
106     public RenderingContext withChildTableAliasCalculator(TableAliasCalculator childTableAliasCalculator) {
107         TableAliasCalculator tac = new TableAliasCalculatorWithParent.Builder()
108                 .withParent(tableAliasCalculator)
109                 .withChild(childTableAliasCalculator)
110                 .build();
111 
112         return new Builder()
113                 .withRenderingStrategy(this.renderingStrategy)
114                 .withSequence(this.sequence)
115                 .withParameterName(this.configuredParameterName)
116                 .withTableAliasCalculator(tac)
117                 .build();
118     }
119 
120     public static Builder withRenderingStrategy(RenderingStrategy renderingStrategy) {
121         return new Builder().withRenderingStrategy(renderingStrategy);
122     }
123 
124     public static class Builder {
125         private RenderingStrategy renderingStrategy;
126         private AtomicInteger sequence;
127         private TableAliasCalculator tableAliasCalculator = TableAliasCalculator.empty();
128         private String parameterName;
129 
130         public Builder withRenderingStrategy(RenderingStrategy renderingStrategy) {
131             this.renderingStrategy = renderingStrategy;
132             return this;
133         }
134 
135         public Builder withSequence(AtomicInteger sequence) {
136             this.sequence = sequence;
137             return this;
138         }
139 
140         public Builder withTableAliasCalculator(TableAliasCalculator tableAliasCalculator) {
141             this.tableAliasCalculator = tableAliasCalculator;
142             return this;
143         }
144 
145         public Builder withParameterName(String parameterName) {
146             this.parameterName = parameterName;
147             return this;
148         }
149 
150         public RenderingContext build() {
151             return new RenderingContext(this);
152         }
153     }
154 }
