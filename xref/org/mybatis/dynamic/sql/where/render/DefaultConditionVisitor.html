16  package org.mybatis.dynamic.sql.where.render;
17  
18  import static org.mybatis.dynamic.sql.util.StringUtilities.spaceBefore;
19  
20  import java.util.Objects;
21  import java.util.stream.Collectors;
22  
23  import org.mybatis.dynamic.sql.AbstractColumnComparisonCondition;
24  import org.mybatis.dynamic.sql.AbstractListValueCondition;
25  import org.mybatis.dynamic.sql.AbstractNoValueCondition;
26  import org.mybatis.dynamic.sql.AbstractSingleValueCondition;
27  import org.mybatis.dynamic.sql.AbstractSubselectCondition;
28  import org.mybatis.dynamic.sql.AbstractTwoValueCondition;
29  import org.mybatis.dynamic.sql.BindableColumn;
30  import org.mybatis.dynamic.sql.ConditionVisitor;
31  import org.mybatis.dynamic.sql.render.RenderedParameterInfo;
32  import org.mybatis.dynamic.sql.render.RenderingContext;
33  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
34  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
35  import org.mybatis.dynamic.sql.util.FragmentCollector;
36  
37  public class DefaultConditionVisitor<T> implements ConditionVisitor<T, FragmentAndParameters> {
38  
39      private final BindableColumn<T> column;
40      private final RenderingContext renderingContext;
41  
42      private DefaultConditionVisitor(Builder<T> builder) {
43          column = Objects.requireNonNull(builder.column);
44          renderingContext = Objects.requireNonNull(builder.renderingContext);
45      }
46  
47      @Override
48      public FragmentAndParameters visit(AbstractListValueCondition<T> condition) {
49          FragmentCollector fc = condition.values()
50                  .map(this::toFragmentAndParameters)
51                  .collect(FragmentCollector.collect());
52  
53          String joinedFragments =
54                  fc.collectFragments(Collectors.joining(",", "(", ")")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
55          String finalFragment = condition.operator()
56                  + spaceBefore(joinedFragments);
57  
58          return FragmentAndParameters
59                  .withFragment(finalFragment)
60                  .withParameters(fc.parameters())
61                  .build();
62      }
63  
64      @Override
65      public FragmentAndParameters visit(AbstractNoValueCondition<T> condition) {
66          return FragmentAndParameters.fromFragment(condition.operator());
67      }
68  
69      @Override
70      public FragmentAndParameters visit(AbstractSingleValueCondition<T> condition) {
71          RenderedParameterInfo parameterInfo = renderingContext.calculateParameterInfo(column);
72          String finalFragment = condition.operator()
73                  + spaceBefore(parameterInfo.renderedPlaceHolder());
74  
75          return FragmentAndParameters.withFragment(finalFragment)
76                  .withParameter(parameterInfo.parameterMapKey(), convertValue(condition.value()))
77                  .build();
78      }
79  
80      @Override
81      public FragmentAndParameters visit(AbstractTwoValueCondition<T> condition) {
82          RenderedParameterInfo parameterInfo1 = renderingContext.calculateParameterInfo(column);
83          RenderedParameterInfo parameterInfo2 = renderingContext.calculateParameterInfo(column);
84  
85          String finalFragment = condition.operator1()
86                  + spaceBefore(parameterInfo1.renderedPlaceHolder())
87                  + spaceBefore(condition.operator2())
88                  + spaceBefore(parameterInfo2.renderedPlaceHolder());
89  
90          return FragmentAndParameters.withFragment(finalFragment)
91                  .withParameter(parameterInfo1.parameterMapKey(), convertValue(condition.value1()))
92                  .withParameter(parameterInfo2.parameterMapKey(), convertValue(condition.value2()))
93                  .build();
94      }
95  
96      @Override
97      public FragmentAndParameters visit(AbstractSubselectCondition<T> condition) {
98          SelectStatementProvider selectStatement = condition.selectModel().render(renderingContext);
99  
100         String finalFragment = condition.operator()
101                 + " (" //$NON-NLS-1$
102                 + selectStatement.getSelectStatement()
103                 + ")"; //$NON-NLS-1$
104 
105         return FragmentAndParameters.withFragment(finalFragment)
106                 .withParameters(selectStatement.getParameters())
107                 .build();
108     }
109 
110     @Override
111     public FragmentAndParameters visit(AbstractColumnComparisonCondition<T> condition) {
112         FragmentAndParameters renderedRightColumn = condition.rightColumn().render(renderingContext);
113         String finalFragment = condition.operator()
114                 + spaceBefore(renderedRightColumn.fragment());
115         return FragmentAndParameters.withFragment(finalFragment)
116                 .withParameters(renderedRightColumn.parameters())
117                 .build();
118     }
119 
120     private Object convertValue(T value) {
121         return column.convertParameterType(value);
122     }
123 
124     private FragmentAndParameters toFragmentAndParameters(T value) {
125         RenderedParameterInfo parameterInfo = renderingContext.calculateParameterInfo(column);
126         return FragmentAndParameters.withFragment(parameterInfo.renderedPlaceHolder())
127                 .withParameter(parameterInfo.parameterMapKey(), convertValue(value))
128                 .build();
129     }
130 
131     public static <T> Builder<T> withColumn(BindableColumn<T> column) {
132         return new Builder<T>().withColumn(column);
133     }
134 
135     public static class Builder<T> {
136         private BindableColumn<T> column;
137         private RenderingContext renderingContext;
138 
139         public Builder<T> withColumn(BindableColumn<T> column) {
140             this.column = column;
141             return this;
142         }
143 
144         public Builder<T> withRenderingContext(RenderingContext renderingContext) {
145             this.renderingContext = renderingContext;
146             return this;
147         }
148 
149         public DefaultConditionVisitor<T> build() {
150             return new DefaultConditionVisitor<>(this);
151         }
152     }
153 }
