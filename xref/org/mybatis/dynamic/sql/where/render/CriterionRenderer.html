40  /**
41   * Renders a {@link SqlCriterion} to a {@link RenderedCriterion}. The process is complex because all conditions
42   * may or may not be a candidate for rendering. For example, "isEqualWhenPresent" will not render when the value
43   * is null. It is also complex because SqlCriterion may or may not include sub-criteria.
44   *
45   * <p>Rendering is a recursive process. The renderer will recurse into each sub-criteria - which may also
46   * contain further sub-criteria - until all possible sub-criteria are rendered into a single fragment. So, for example,
47   * the fragment may end up looking like:
48   *
49   * <pre>
50   *     col1 = ? and (col2 = ? or (col3 = ? and col4 = ?))
51   * </pre>
52   *
53   * <p>It is also possible that the end result will be empty if all criteria and sub-criteria are not valid for
54   * rendering.
55   *
56   * @author Jeff Butler
57   */
58  public class CriterionRenderer implements SqlCriterionVisitor<Optional<RenderedCriterion>> {
59      private final AtomicInteger sequence;
60      private final RenderingStrategy renderingStrategy;
61      private final TableAliasCalculator tableAliasCalculator;
62      private final String parameterName;
63  
64      private CriterionRenderer(Builder builder) {
65          sequence = Objects.requireNonNull(builder.sequence);
66          renderingStrategy = Objects.requireNonNull(builder.renderingStrategy);
67          tableAliasCalculator = Objects.requireNonNull(builder.tableAliasCalculator);
68          parameterName = builder.parameterName;
69      }
70  
71      @Override
72      public <T> Optional<RenderedCriterion> visit(ColumnAndConditionCriterion<T> criterion) {
73          Optional<FragmentAndParameters> initialCriterion = renderColumnAndCondition(criterion);
74          List<RenderedCriterion> renderedSubCriteria = renderSubCriteria(criterion.subCriteria());
75  
76          return initialCriterion.map(fp -> calculateRenderedCriterion(fp, renderedSubCriteria, this::calculateFragment))
77                  .orElseGet(() -> calculateRenderedCriterion(renderedSubCriteria, this::calculateFragment));
78      }
79  
80      @Override
81      public Optional<RenderedCriterion> visit(ExistsCriterion criterion) {
82          FragmentAndParameters initialCriterion = renderExists(criterion);
83          List<RenderedCriterion> renderedSubCriteria = renderSubCriteria(criterion.subCriteria());
84  
85          return calculateRenderedCriterion(initialCriterion, renderedSubCriteria, this::calculateFragment);
86      }
87  
88      @Override
89      public Optional<RenderedCriterion> visit(CriteriaGroup criterion) {
90          return renderCriteriaGroup(criterion, this::calculateFragment);
91      }
92  
93      @Override
94      public Optional<RenderedCriterion> visit(NotCriterion criterion) {
95          return renderCriteriaGroup(criterion, this::calculateNotFragment);
96      }
97  
98      private Optional<RenderedCriterion> renderCriteriaGroup(CriteriaGroup criterion,
99                                                              Function<FragmentCollector, String> fragmentCalculator) {
100         return criterion.initialCriterion().map(ic -> render(ic, criterion.subCriteria(), fragmentCalculator))
101                 .orElseGet(() -> render(criterion.subCriteria(), fragmentCalculator));
102     }
103 
104     public Optional<RenderedCriterion> render(SqlCriterion initialCriterion, List<AndOrCriteriaGroup> subCriteria,
105                                               Function<FragmentCollector, String> fragmentCalculator) {
106         Optional<FragmentAndParameters> fragmentAndParameters = initialCriterion.accept(this)
107                 .map(RenderedCriterion::fragmentAndParameters);
108         List<RenderedCriterion> renderedSubCriteria = renderSubCriteria(subCriteria);
109 
110         return fragmentAndParameters.map(fp -> calculateRenderedCriterion(fp, renderedSubCriteria, fragmentCalculator))
111                 .orElseGet(() -> calculateRenderedCriterion(renderedSubCriteria, fragmentCalculator));
112     }
113 
114     public Optional<RenderedCriterion> render(List<AndOrCriteriaGroup> subCriteria,
115                                               Function<FragmentCollector, String> fragmentCalculator) {
116         List<RenderedCriterion> renderedSubCriteria = renderSubCriteria(subCriteria);
117         return calculateRenderedCriterion(renderedSubCriteria, fragmentCalculator);
118     }
119 
120     private <T> Optional<FragmentAndParameters> renderColumnAndCondition(ColumnAndConditionCriterion<T> criterion) {
121         if (criterion.condition().shouldRender()) {
122             return Optional.of(renderCondition(criterion));
123         } else {
124             criterion.condition().renderingSkipped();
125             return Optional.empty();
126         }
127     }
128 
129     private FragmentAndParameters renderExists(ExistsCriterion criterion) {
130         ExistsPredicate existsPredicate = criterion.existsPredicate();
131 
132         SelectStatementProvider selectStatement = SelectRenderer
133                 .withSelectModel(existsPredicate.selectModelBuilder().build())
134                 .withRenderingStrategy(renderingStrategy)
135                 .withSequence(sequence)
136                 .withParentTableAliasCalculator(tableAliasCalculator)
137                 .build()
138                 .render();
139 
140         String fragment = existsPredicate.operator()
141                 + " (" //$NON-NLS-1$
142                 + selectStatement.getSelectStatement()
143                 + ")"; //$NON-NLS-1$
144 
145         return FragmentAndParameters
146                 .withFragment(fragment)
147                 .withParameters(selectStatement.getParameters())
148                 .build();
149     }
150 
151     private List<RenderedCriterion> renderSubCriteria(List<AndOrCriteriaGroup> subCriteria) {
152         return subCriteria.stream().map(this::renderAndOrCriteriaGroup)
153                 .filter(Optional::isPresent)
154                 .map(Optional::get)
155                 .collect(Collectors.toList());
156     }
157 
158     private Optional<RenderedCriterion> renderAndOrCriteriaGroup(AndOrCriteriaGroup criterion) {
159         return criterion.initialCriterion().map(ic -> render(ic, criterion.subCriteria(), this::calculateFragment))
160                 .orElseGet(() -> render(criterion.subCriteria(), this::calculateFragment))
161                 .map(rc -> rc.withConnector(criterion.connector()));
162     }
163 
164     private Optional<RenderedCriterion> calculateRenderedCriterion(FragmentAndParameters initialCriterion,
165             List<RenderedCriterion> renderedSubCriteria, Function<FragmentCollector, String> fragmentCalculator) {
166         return Optional.of(calculateRenderedCriterion(
167                 collectSqlFragments(initialCriterion, renderedSubCriteria), fragmentCalculator));
168     }
169 
170     private RenderedCriterion calculateRenderedCriterion(FragmentCollector fragmentCollector,
171                                                          Function<FragmentCollector, String> fragmentCalculator) {
172         FragmentAndParameters fragmentAndParameters = FragmentAndParameters
173                 .withFragment(fragmentCalculator.apply(fragmentCollector))
174                 .withParameters(fragmentCollector.parameters())
175                 .build();
176 
177         return new RenderedCriterion.Builder()
178                 .withFragmentAndParameters(fragmentAndParameters)
179                 .build();
180     }
181 
182     private Optional<RenderedCriterion> calculateRenderedCriterion(List<RenderedCriterion> renderedSubCriteria,
183             Function<FragmentCollector, String> fragmentCalculator) {
184         return collectSqlFragments(renderedSubCriteria).map(fc -> calculateRenderedCriterion(fc, fragmentCalculator));
185     }
186 
187     private <T> FragmentAndParameters renderCondition(ColumnAndConditionCriterion<T> criterion) {
188         WhereConditionVisitor<T> visitor = WhereConditionVisitor.withColumn(criterion.column())
189                 .withRenderingStrategy(renderingStrategy)
190                 .withSequence(sequence)
191                 .withTableAliasCalculator(tableAliasCalculator)
192                 .withParameterName(parameterName)
193                 .build();
194         return criterion.condition().accept(visitor);
195     }
196 
197     /**
198      * This method encapsulates the logic of building a collection of fragments from an initial condition
199      * and a list of rendered sub criteria. In this overload we know there is an initial condition
200      * and there may be subcriteria. The collector will contain the initial condition and any rendered subcriteria
201      * in order.
202      *
203      * @param initialCondition - may not be null. If there is no initial condition, then use the other overload
204      * @param renderedSubCriteria - a list of previously rendered sub criteria. The sub criteria will all
205      *                            have connectors (either an AND or an OR)
206      * @return a fragment collector whose fragments represent the final calculated list of fragments and parameters.
207      *     The fragment collector can be used to calculate the single composed fragment - either as a where clause, or
208      *     a valid rendered sub criteria in the case of a recursive call.
209      */
210     private FragmentCollector collectSqlFragments(FragmentAndParameters initialCondition,
211                                                   List<RenderedCriterion> renderedSubCriteria) {
212         return renderedSubCriteria.stream()
213                 .map(RenderedCriterion::fragmentAndParametersWithConnector)
214                 .collect(FragmentCollector.collect(initialCondition));
215     }
216 
217     /**
218      * This method encapsulates the logic of building a collection of fragments from a list of rendered sub criteria.
219      * In this overload we take the initial condition to be the first element in the subcriteria list.
220      * The collector will contain the rendered subcriteria in order. However, the connector from the first rendered
221      * sub criterion will be removed. This to avoid generating an invalid where clause like "where and a < 3"
222      *
223      * @param renderedSubCriteria - a list of previously rendered sub criteria. The sub criteria will all
224      *                            have connectors (either an AND or an OR)
225      * @return a fragment collector whose fragments represent the final calculated list of fragments and parameters.
226      *     The fragment collector can be used to calculate the single composed fragment - either as a where clause, or
227      *     a valid rendered sub criteria in the case of a recursive call.
228      */
229     private Optional<FragmentCollector> collectSqlFragments(List<RenderedCriterion> renderedSubCriteria) {
230         if (renderedSubCriteria.isEmpty()) {
231             return Optional.empty();
232         }
233 
234         FragmentAndParameters firstCondition = renderedSubCriteria.get(0).fragmentAndParameters();
235 
236         FragmentCollector fc = renderedSubCriteria.stream()
237                 .skip(1)
238                 .map(RenderedCriterion::fragmentAndParametersWithConnector)
239                 .collect(FragmentCollector.collect(firstCondition));
240 
241         return Optional.of(fc);
242     }
243 
244     private String calculateFragment(FragmentCollector collector) {
245         if (collector.hasMultipleFragments()) {
246             return collector.fragments()
247                     .collect(Collectors.joining(" ", "(", ")")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
248         } else {
249             return collector.fragments().findFirst().orElse(""); //$NON-NLS-1$
250         }
251     }
252 
253     private String calculateNotFragment(FragmentCollector collector) {
254         if (collector.hasMultipleFragments()) {
255             return collector.fragments()
256                     .collect(Collectors.joining(" ", "not (", ")")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
257         } else {
258             return collector.fragments().findFirst().map(s -> "not " + s).orElse(""); //$NON-NLS-1$ //$NON-NLS-2$
259         }
260     }

281         }

284             this.parameterName = parameterName;

286         }

288         public CriterionRenderer build() {
289             return new CriterionRenderer(this);
290         }
291     }
