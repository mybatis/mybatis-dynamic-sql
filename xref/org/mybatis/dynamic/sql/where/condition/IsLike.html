16  package org.mybatis.dynamic.sql.where.condition;
17  
18  import java.util.NoSuchElementException;
19  import java.util.function.Function;
20  import java.util.function.Predicate;
21  
22  import org.jspecify.annotations.NonNull;
23  import org.mybatis.dynamic.sql.AbstractSingleValueCondition;
24  
25  public class IsLike<T> extends AbstractSingleValueCondition<T>
26          implements AbstractSingleValueCondition.Filterable<T>, AbstractSingleValueCondition.Mappable<T> {
27  
28      private static final IsLike<?> EMPTY = new IsLike<Object>(-1) {
29          @Override
30          public Object value() {
31              throw new NoSuchElementException("No value present"); //$NON-NLS-1$
32          }
33  
34          @Override
35          public boolean isEmpty() {
36              return true;
37          }
38      };
39  
40      public static <T> IsLike<T> empty() {
41          @SuppressWarnings("unchecked")
42          IsLike<T> t = (IsLike<T>) EMPTY;
43          return t;
44      }
45  
46      protected IsLike(T value) {
47          super(value);
48      }
49  
50      @Override
51      public String operator() {
52          return "like"; //$NON-NLS-1$
53      }
54  
55      public static <T> IsLike<T> of(T value) {
56          return new IsLike<>(value);
57      }
58  
59      @Override
60      public IsLike<T> filter(Predicate<? super @NonNull T> predicate) {
61          return filterSupport(predicate, IsLike::empty, this);
62      }
63  
64      @Override
65      public <R> IsLike<R> map(Function<? super @NonNull T, ? extends @NonNull R> mapper) {
66          return mapSupport(mapper, IsLike::new, IsLike::empty);
67      }
68  }
