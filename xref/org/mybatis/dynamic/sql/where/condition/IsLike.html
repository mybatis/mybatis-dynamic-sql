16  package org.mybatis.dynamic.sql.where.condition;
17  
18  import java.util.function.Function;
19  import java.util.function.Predicate;
20  
21  import org.mybatis.dynamic.sql.AbstractSingleValueCondition;
22  
23  public class IsLike<T> extends AbstractSingleValueCondition<T> {
24      private static final IsLike<?> EMPTY = new IsLike<Object>(null) {
25          @Override
26          public boolean shouldRender() {
27              return false;
28          }
29      };
30  
31      public static <T> IsLike<T> empty() {
32          @SuppressWarnings("unchecked")
33          IsLike<T> t = (IsLike<T>) EMPTY;
34          return t;
35      }
36  
37      protected IsLike(T value) {
38          super(value);
39      }
40  
41      @Override
42      public String renderCondition(String columnName, String placeholder) {
43          return columnName + " like " + placeholder; //$NON-NLS-1$
44      }
45  
46      public static <T> IsLike<T> of(T value) {
47          return new IsLike<>(value);
48      }
49  
50      @Override
51      public IsLike<T> filter(Predicate<? super T> predicate) {
52          return filterSupport(predicate, IsLike::empty, this);
53      }
54  
55      /**
56       * If renderable, apply the mapping to the value and return a new condition with the new value. Else return a
57       *     condition that will not render (this).
58       *
59       * @param mapper a mapping function to apply to the value, if renderable
60       * @param <R> type of the new condition
61       * @return a new condition with the result of applying the mapper to the value of this condition,
62       *     if renderable, otherwise a condition that will not render.
63       */
64      public <R> IsLike<R> map(Function<? super T, ? extends R> mapper) {
65          return mapSupport(mapper, IsLike::new, IsLike::empty);
66      }
67  }
