16  package org.mybatis.dynamic.sql.insert.render;
17  
18  import java.util.ArrayList;
19  import java.util.Collections;
20  import java.util.List;
21  import java.util.Objects;
22  import java.util.stream.Collectors;
23  
24  public class BatchInsert<T> {
25      private final String insertStatement;
26      private final List<T> records;
27  
28      private BatchInsert(Builder<T> builder) {
29          insertStatement = Objects.requireNonNull(builder.insertStatement);
30          records = Collections.unmodifiableList(Objects.requireNonNull(builder.records));
31      }
32  
33      /**
34       * Returns a list of InsertStatement objects. This is useful for MyBatis batch support.
35       *
36       * @return a List of InsertStatements
37       */
38      public List<InsertStatementProvider<T>> insertStatements() {
39          return records.stream()
40                  .map(this::toInsertStatement)
41                  .collect(Collectors.toList());
42      }
43  
44      private InsertStatementProvider<T> toInsertStatement(T row) {
45          return DefaultInsertStatementProvider.withRow(row)
46                  .withInsertStatement(insertStatement)
47                  .build();
48      }
49  
50      /**
51       * Returns the generated SQL for this batch. This is useful for Spring JDBC batch support.
52       *
53       * @return the generated INSERT statement
54       */
55      public String getInsertStatementSQL() {
56          return insertStatement;
57      }
58  
59      public List<T> getRecords() {
60          return Collections.unmodifiableList(records);
61      }
62  
63      public static <T> Builder<T> withRecords(List<T> records) {
64          return new Builder<T>().withRecords(records);
65      }
66  
67      public static class Builder<T> {
68          private String insertStatement;
69          private final List<T> records = new ArrayList<>();
70  
71          public Builder<T> withInsertStatement(String insertStatement) {
72              this.insertStatement = insertStatement;
73              return this;
74          }
75  
76          public Builder<T> withRecords(List<T> records) {
77              this.records.addAll(records);
78              return this;
79          }
80  
81          public BatchInsert<T> build() {
82              return new BatchInsert<>(this);
83          }
84      }
85  }
