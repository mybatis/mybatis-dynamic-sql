16  package org.mybatis.dynamic.sql.insert.render;
17  
18  import java.util.Optional;
19  
20  import org.mybatis.dynamic.sql.SqlColumn;
21  import org.mybatis.dynamic.sql.render.RenderingStrategy;
22  import org.mybatis.dynamic.sql.util.ConstantMapping;
23  import org.mybatis.dynamic.sql.util.InsertMappingVisitor;
24  import org.mybatis.dynamic.sql.util.NullMapping;
25  import org.mybatis.dynamic.sql.util.PropertyMapping;
26  import org.mybatis.dynamic.sql.util.PropertyWhenPresentMapping;
27  import org.mybatis.dynamic.sql.util.StringConstantMapping;
28  
29  public class ValuePhraseVisitor extends InsertMappingVisitor<Optional<FieldAndValue>> {
30  
31      protected final RenderingStrategy renderingStrategy;
32  
33      public ValuePhraseVisitor(RenderingStrategy renderingStrategy) {
34          this.renderingStrategy = renderingStrategy;
35      }
36  
37      @Override
38      public Optional<FieldAndValue> visit(NullMapping mapping) {
39          return FieldAndValue.withFieldName(mapping.columnName())
40                  .withValuePhrase("null") //$NON-NLS-1$
41                  .buildOptional();
42      }
43  
44      @Override
45      public Optional<FieldAndValue> visit(ConstantMapping mapping) {
46          return FieldAndValue.withFieldName(mapping.columnName())
47                  .withValuePhrase(mapping.constant())
48                  .buildOptional();
49      }
50  
51      @Override
52      public Optional<FieldAndValue> visit(StringConstantMapping mapping) {
53          return FieldAndValue.withFieldName(mapping.columnName())
54                  .withValuePhrase("'" + mapping.constant() + "'") //$NON-NLS-1$ //$NON-NLS-2$
55                  .buildOptional();
56      }
57  
58      @Override
59      public Optional<FieldAndValue> visit(PropertyMapping mapping) {
60          return FieldAndValue.withFieldName(mapping.columnName())
61                  .withValuePhrase(mapping.mapColumn(c -> calculateJdbcPlaceholder(c, mapping.property())))
62                  .buildOptional();
63      }
64  
65      @Override
66      public Optional<FieldAndValue> visit(PropertyWhenPresentMapping mapping) {
67          if (mapping.shouldRender()) {
68              return visit((PropertyMapping) mapping);
69          } else {
70              return Optional.empty();
71          }
72      }
73  
74      private String calculateJdbcPlaceholder(SqlColumn<?> column, String parameterName) {
75          return column.renderingStrategy().orElse(renderingStrategy)
76                  .getFormattedJdbcPlaceholder(column, "row", parameterName); //$NON-NLS-1$
77      }
78  }
