16  package org.mybatis.dynamic.sql.update.render;
17  
18  import java.util.Objects;
19  import java.util.Optional;
20  import java.util.concurrent.atomic.AtomicInteger;
21  import java.util.function.Function;
22  
23  import org.mybatis.dynamic.sql.SqlColumn;
24  import org.mybatis.dynamic.sql.render.RenderingStrategy;
25  import org.mybatis.dynamic.sql.render.TableAliasCalculator;
26  import org.mybatis.dynamic.sql.select.render.SelectRenderer;
27  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
28  import org.mybatis.dynamic.sql.util.AbstractColumnMapping;
29  import org.mybatis.dynamic.sql.util.ColumnToColumnMapping;
30  import org.mybatis.dynamic.sql.util.ConstantMapping;
31  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
32  import org.mybatis.dynamic.sql.util.NullMapping;
33  import org.mybatis.dynamic.sql.util.SelectMapping;
34  import org.mybatis.dynamic.sql.util.StringConstantMapping;
35  import org.mybatis.dynamic.sql.util.UpdateMappingVisitor;
36  import org.mybatis.dynamic.sql.util.ValueMapping;
37  import org.mybatis.dynamic.sql.util.ValueOrNullMapping;
38  import org.mybatis.dynamic.sql.util.ValueWhenPresentMapping;
39  
40  public class SetPhraseVisitor extends UpdateMappingVisitor<Optional<FragmentAndParameters>> {
41  
42      private final AtomicInteger sequence;
43      private final RenderingStrategy renderingStrategy;
44      private final Function<SqlColumn<?>, String> aliasedColumnNameFunction;
45  
46      public SetPhraseVisitor(AtomicInteger sequence, RenderingStrategy renderingStrategy,
47                              TableAliasCalculator tableAliasCalculator) {
48          this.sequence = Objects.requireNonNull(sequence);
49          this.renderingStrategy = Objects.requireNonNull(renderingStrategy);
50          Objects.requireNonNull(tableAliasCalculator);
51          aliasedColumnNameFunction = c -> tableAliasCalculator.aliasForColumn(c.table())
52                  .map(alias -> alias + "." + c.name())  //$NON-NLS-1$
53                  .orElseGet(c::name);
54      }
55  
56      @Override
57      public Optional<FragmentAndParameters> visit(NullMapping mapping) {
58          return FragmentAndParameters
59                  .withFragment(mapping.mapColumn(aliasedColumnNameFunction) + " = null") //$NON-NLS-1$
60                  .buildOptional();
61      }
62  
63      @Override
64      public Optional<FragmentAndParameters> visit(ConstantMapping mapping) {
65          String fragment = mapping.mapColumn(aliasedColumnNameFunction) + " = " + mapping.constant(); //$NON-NLS-1$
66          return FragmentAndParameters.withFragment(fragment)
67                  .buildOptional();
68      }
69  
70      @Override
71      public Optional<FragmentAndParameters> visit(StringConstantMapping mapping) {
72          String fragment = mapping.mapColumn(aliasedColumnNameFunction)
73                  + " = '" //$NON-NLS-1$
74                  + mapping.constant()
75                  + "'"; //$NON-NLS-1$
76  
77          return FragmentAndParameters.withFragment(fragment)
78                  .buildOptional();
79      }
80  
81      @Override
82      public <T> Optional<FragmentAndParameters> visit(ValueMapping<T> mapping) {
83          return buildFragment(mapping, mapping.value());
84      }
85  
86      @Override
87      public <T> Optional<FragmentAndParameters> visit(ValueOrNullMapping<T> mapping) {
88          return mapping.value()
89                  .map(v -> buildFragment(mapping, v))
90                  .orElseGet(() -> FragmentAndParameters
91                          .withFragment(mapping.mapColumn(aliasedColumnNameFunction) + " = null") //$NON-NLS-1$
92                          .buildOptional()
93                  );
94      }
95  
96      @Override
97      public <T> Optional<FragmentAndParameters> visit(ValueWhenPresentMapping<T> mapping) {
98          return mapping.value().flatMap(v -> buildFragment(mapping, v));
99      }
100 
101     @Override
102     public Optional<FragmentAndParameters> visit(SelectMapping mapping) {
103         SelectStatementProvider selectStatement = SelectRenderer.withSelectModel(mapping.selectModel())
104                 .withRenderingStrategy(renderingStrategy)
105                 .withSequence(sequence)
106                 .build()
107                 .render();
108 
109         String fragment = mapping.mapColumn(aliasedColumnNameFunction)
110                 + " = (" //$NON-NLS-1$
111                 + selectStatement.getSelectStatement()
112                 + ")"; //$NON-NLS-1$
113 
114         return FragmentAndParameters.withFragment(fragment)
115                 .withParameters(selectStatement.getParameters())
116                 .buildOptional();
117     }
118 
119     @Override
120     public Optional<FragmentAndParameters> visit(ColumnToColumnMapping mapping) {
121         String setPhrase = mapping.mapColumn(aliasedColumnNameFunction)
122                 + " = "  //$NON-NLS-1$
123                 + mapping.rightColumn().renderWithTableAlias(TableAliasCalculator.empty());
124 
125         return FragmentAndParameters.withFragment(setPhrase)
126                 .buildOptional();
127     }
128 
129     private <T> Optional<FragmentAndParameters> buildFragment(AbstractColumnMapping mapping, T value) {
130         String mapKey = RenderingStrategy.formatParameterMapKey(sequence);
131 
132         String jdbcPlaceholder = mapping.mapColumn(c -> calculateJdbcPlaceholder(c, mapKey));
133         String setPhrase = mapping.mapColumn(aliasedColumnNameFunction)
134                 + " = "  //$NON-NLS-1$
135                 + jdbcPlaceholder;
136 
137         return FragmentAndParameters.withFragment(setPhrase)
138                 .withParameter(mapKey, value)
139                 .buildOptional();
140     }
141 
142     private String calculateJdbcPlaceholder(SqlColumn<?> column, String parameterName) {
143         return column.renderingStrategy().orElse(renderingStrategy)
144                 .getFormattedJdbcPlaceholder(column, RenderingStrategy.DEFAULT_PARAMETER_PREFIX, parameterName);
145     }
146 }
