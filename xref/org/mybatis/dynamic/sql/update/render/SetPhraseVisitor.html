16  package org.mybatis.dynamic.sql.update.render;
17  
18  import java.util.Objects;
19  import java.util.Optional;
20  
21  import org.mybatis.dynamic.sql.render.RenderedParameterInfo;
22  import org.mybatis.dynamic.sql.render.RenderingContext;
23  import org.mybatis.dynamic.sql.select.render.SelectStatementProvider;
24  import org.mybatis.dynamic.sql.util.AbstractColumnMapping;
25  import org.mybatis.dynamic.sql.util.ColumnToColumnMapping;
26  import org.mybatis.dynamic.sql.util.ConstantMapping;
27  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
28  import org.mybatis.dynamic.sql.util.NullMapping;
29  import org.mybatis.dynamic.sql.util.SelectMapping;
30  import org.mybatis.dynamic.sql.util.StringConstantMapping;
31  import org.mybatis.dynamic.sql.util.StringUtilities;
32  import org.mybatis.dynamic.sql.util.UpdateMappingVisitor;
33  import org.mybatis.dynamic.sql.util.ValueMapping;
34  import org.mybatis.dynamic.sql.util.ValueOrNullMapping;
35  import org.mybatis.dynamic.sql.util.ValueWhenPresentMapping;
36  
37  public class SetPhraseVisitor extends UpdateMappingVisitor<Optional<FragmentAndParameters>> {
38  
39      private final RenderingContext renderingContext;
40  
41      public SetPhraseVisitor(RenderingContext renderingContext) {
42          this.renderingContext = Objects.requireNonNull(renderingContext);
43      }
44  
45      @Override
46      public Optional<FragmentAndParameters> visit(NullMapping mapping) {
47          return buildNullFragment(mapping);
48      }
49  
50      @Override
51      public Optional<FragmentAndParameters> visit(ConstantMapping mapping) {
52          String fragment = renderingContext.aliasedColumnName(mapping.column())
53                  + " = " + mapping.constant(); //$NON-NLS-1$
54          return FragmentAndParameters.withFragment(fragment)
55                  .buildOptional();
56      }
57  
58      @Override
59      public Optional<FragmentAndParameters> visit(StringConstantMapping mapping) {
60          String fragment = renderingContext.aliasedColumnName(mapping.column())
61                  + " = " //$NON-NLS-1$
62                  + StringUtilities.formatConstantForSQL(mapping.constant());
63  
64          return FragmentAndParameters.withFragment(fragment)
65                  .buildOptional();
66      }
67  
68      @Override
69      public <T> Optional<FragmentAndParameters> visit(ValueMapping<T> mapping) {
70          return buildValueFragment(mapping, mapping.value());
71      }
72  
73      @Override
74      public <T> Optional<FragmentAndParameters> visit(ValueOrNullMapping<T> mapping) {
75          return mapping.value()
76                  .map(v -> buildValueFragment(mapping, v))
77                  .orElseGet(() -> buildNullFragment(mapping));
78      }
79  
80      @Override
81      public <T> Optional<FragmentAndParameters> visit(ValueWhenPresentMapping<T> mapping) {
82          return mapping.value().flatMap(v -> buildValueFragment(mapping, v));
83      }
84  
85      @Override
86      public Optional<FragmentAndParameters> visit(SelectMapping mapping) {
87          SelectStatementProvider selectStatement = mapping.selectModel().render(renderingContext);
88          String fragment = renderingContext.aliasedColumnName(mapping.column())
89                  + " = (" //$NON-NLS-1$
90                  + selectStatement.getSelectStatement()
91                  + ")"; //$NON-NLS-1$
92  
93          return FragmentAndParameters.withFragment(fragment)
94                  .withParameters(selectStatement.getParameters())
95                  .buildOptional();
96      }
97  
98      @Override
99      public Optional<FragmentAndParameters> visit(ColumnToColumnMapping mapping) {
100         FragmentAndParameters renderedColumn = mapping.rightColumn().render(renderingContext);
101 
102         String setPhrase = renderingContext.aliasedColumnName(mapping.column())
103                 + " = "  //$NON-NLS-1$
104                 + renderedColumn.fragment();
105 
106         return FragmentAndParameters.withFragment(setPhrase)
107                 .withParameters(renderedColumn.parameters())
108                 .buildOptional();
109     }
110 
111     private <T> Optional<FragmentAndParameters> buildValueFragment(AbstractColumnMapping mapping, T value) {
112         RenderedParameterInfo parameterInfo = renderingContext.calculateParameterInfo(mapping.column());
113         String setPhrase = renderingContext.aliasedColumnName(mapping.column())
114                 + " = "  //$NON-NLS-1$
115                 + parameterInfo.renderedPlaceHolder();
116 
117         return FragmentAndParameters.withFragment(setPhrase)
118                 .withParameter(parameterInfo.parameterMapKey(), value)
119                 .buildOptional();
120     }
121 
122     private Optional<FragmentAndParameters> buildNullFragment(AbstractColumnMapping mapping) {
123         return FragmentAndParameters
124                 .withFragment(renderingContext.aliasedColumnName(mapping.column()) + " = null") //$NON-NLS-1$
125                 .buildOptional();
126     }
127 }
