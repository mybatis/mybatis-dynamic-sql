16  package org.mybatis.dynamic.sql.update.render;
17  
18  import java.util.List;
19  import java.util.Objects;
20  import java.util.Optional;
21  import java.util.stream.Collectors;
22  
23  import org.mybatis.dynamic.sql.common.OrderByModel;
24  import org.mybatis.dynamic.sql.common.OrderByRenderer;
25  import org.mybatis.dynamic.sql.configuration.StatementConfiguration;
26  import org.mybatis.dynamic.sql.render.ExplicitTableAliasCalculator;
27  import org.mybatis.dynamic.sql.render.RenderedParameterInfo;
28  import org.mybatis.dynamic.sql.render.RenderingContext;
29  import org.mybatis.dynamic.sql.render.RenderingStrategy;
30  import org.mybatis.dynamic.sql.render.TableAliasCalculator;
31  import org.mybatis.dynamic.sql.update.UpdateModel;
32  import org.mybatis.dynamic.sql.util.FragmentAndParameters;
33  import org.mybatis.dynamic.sql.util.FragmentCollector;
34  import org.mybatis.dynamic.sql.util.Validator;
35  import org.mybatis.dynamic.sql.where.EmbeddedWhereModel;
36  
37  public class UpdateRenderer {
38      private final UpdateModel updateModel;
39      private final RenderingContext renderingContext;
40      private final SetPhraseVisitor visitor;
41  
42      private UpdateRenderer(Builder builder) {
43          updateModel = Objects.requireNonNull(builder.updateModel);
44          TableAliasCalculator tableAliasCalculator = builder.updateModel.tableAlias()
45                  .map(a -> ExplicitTableAliasCalculator.of(updateModel.table(), a))
46                  .orElseGet(TableAliasCalculator::empty);
47          renderingContext = RenderingContext
48                  .withRenderingStrategy(Objects.requireNonNull(builder.renderingStrategy))
49                  .withTableAliasCalculator(tableAliasCalculator)
50                  .withStatementConfiguration(builder.statementConfiguration)
51                  .build();
52          visitor = new SetPhraseVisitor(renderingContext);
53      }
54  
55      public UpdateStatementProvider render() {
56          FragmentCollector fragmentCollector = new FragmentCollector();
57  
58          fragmentCollector.add(calculateUpdateStatementStart());
59          fragmentCollector.add(calculateSetPhrase());
60          calculateWhereClause().ifPresent(fragmentCollector::add);
61          calculateOrderByClause().ifPresent(fragmentCollector::add);
62          calculateLimitClause().ifPresent(fragmentCollector::add);
63  
64          return toUpdateStatementProvider(fragmentCollector);
65      }
66  
67      private UpdateStatementProvider toUpdateStatementProvider(FragmentCollector fragmentCollector) {
68          return DefaultUpdateStatementProvider
69                  .withUpdateStatement(fragmentCollector.collectFragments(Collectors.joining(" "))) //$NON-NLS-1$
70                  .withParameters(fragmentCollector.parameters())
71                  .build();
72      }
73  
74      private FragmentAndParameters calculateUpdateStatementStart() {
75          String aliasedTableName = renderingContext.aliasedTableName(updateModel.table());
76          return FragmentAndParameters.fromFragment("update " + aliasedTableName); //$NON-NLS-1$
77      }
78  
79      private FragmentAndParameters calculateSetPhrase() {
80          List<Optional<FragmentAndParameters>> fragmentsAndParameters = updateModel.columnMappings()
81                          .map(m -> m.accept(visitor))
82                          .collect(Collectors.toList());
83  
84          Validator.assertFalse(fragmentsAndParameters.stream().noneMatch(Optional::isPresent),
85                  "ERROR.18"); //$NON-NLS-1$
86  
87          FragmentCollector fragmentCollector = fragmentsAndParameters.stream()
88                  .filter(Optional::isPresent)
89                  .map(Optional::get)
90                  .collect(FragmentCollector.collect());
91  
92          return toSetPhrase(fragmentCollector);
93      }
94  
95      private FragmentAndParameters toSetPhrase(FragmentCollector fragmentCollector) {
96          return fragmentCollector.toFragmentAndParameters(
97                  Collectors.joining(", ", "set ", "")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
98      }
99  
100     private Optional<FragmentAndParameters> calculateWhereClause() {
101         return updateModel.whereModel().flatMap(this::renderWhereClause);
102     }
103 
104     private Optional<FragmentAndParameters> renderWhereClause(EmbeddedWhereModel whereModel) {
105         return whereModel.render(renderingContext);
106     }
107 
108     private Optional<FragmentAndParameters> calculateLimitClause() {
109         return updateModel.limit().map(this::renderLimitClause);
110     }
111 
112     private FragmentAndParameters renderLimitClause(Long limit) {
113         RenderedParameterInfo parameterInfo = renderingContext.calculateParameterInfo();
114 
115         return FragmentAndParameters.withFragment("limit " + parameterInfo.renderedPlaceHolder()) //$NON-NLS-1$
116                 .withParameter(parameterInfo.parameterMapKey(), limit)
117                 .build();
118     }
119 
120     private Optional<FragmentAndParameters> calculateOrderByClause() {
121         return updateModel.orderByModel().map(this::renderOrderByClause);
122     }
123 
124     private FragmentAndParameters renderOrderByClause(OrderByModel orderByModel) {
125         return new OrderByRenderer().render(orderByModel);
126     }
127 
128     public static Builder withUpdateModel(UpdateModel updateModel) {
129         return new Builder().withUpdateModel(updateModel);
130     }
131 
132     public static class Builder {
133         private UpdateModel updateModel;
134         private RenderingStrategy renderingStrategy;
135         private StatementConfiguration statementConfiguration;
136 
137         public Builder withUpdateModel(UpdateModel updateModel) {
138             this.updateModel = updateModel;
139             return this;
140         }
141 
142         public Builder withRenderingStrategy(RenderingStrategy renderingStrategy) {
143             this.renderingStrategy = renderingStrategy;
144             return this;
145         }
146 
147         public Builder withStatementConfiguration(StatementConfiguration statementConfiguration) {
148             this.statementConfiguration = statementConfiguration;
149             return this;
150         }
151 
152         public UpdateRenderer build() {
153             return new UpdateRenderer(this);
154         }
155     }
